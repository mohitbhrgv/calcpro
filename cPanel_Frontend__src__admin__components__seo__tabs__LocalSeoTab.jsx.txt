import React, { useState, useEffect } from "react";
import SafeIcon from "../../../../common/SafeIcon";
import * as FiIcons from "react-icons/fi";
import toast from "react-hot-toast";

const { FiCode, FiClipboard, FiInfo, FiCheckCircle, FiAlertTriangle, FiExternalLink } = FiIcons;

const LocalSeoTab = ({ settings, handleCustomChange }) => {
  const [jsonInput, setJsonInput] = useState("");
  const [isValid, setIsValid] = useState(true);
  const [errorMessage, setErrorMessage] = useState("");

  // Initialize from settings
  useEffect(() => {
    if (settings.seo_local_json_ld) {
      setJsonInput(settings.seo_local_json_ld);
      validateJson(settings.seo_local_json_ld);
    } else {
      const defaultTemplate = JSON.stringify({
        "@context": "https://schema.org",
        "@type": "Organization",
        "name": "%sitename%",
        "url": "%siteurl%",
        "logo": "%sitelogo%",
        "sameAs": [
          "https://www.facebook.com/your-page",
          "https://twitter.com/your-handle"
        ]
      }, null, 2);
      setJsonInput(defaultTemplate);
    }
  }, [settings.seo_local_json_ld]);

  const validateJson = (value) => {
    if (!value || value.trim() === "") {
        setIsValid(true); 
        setErrorMessage("");
        return true;
    }
    try {
      JSON.parse(value);
      setIsValid(true);
      setErrorMessage("");
      return true;
    } catch (e) {
      setIsValid(false);
      // Simplify error message for UI
      const shortError = e.message.split(',')[0]; 
      setErrorMessage(shortError);
      return false;
    }
  };

  const handleChange = (e) => {
    let val = e.target.value;

    // --- SMART CLEANER ---
    // If user pastes <script> tags, strip them out automatically
    if (val.includes("<script")) {
        const cleaned = val.replace(/<script[^>]*>([\s\S]*?)<\/script>/gi, "$1").trim();
        if (cleaned !== val) {
            val = cleaned;
            toast.success("Removed <script> tags automatically.", { icon: 'ðŸ§¹' });
        }
    }

    setJsonInput(val);
    const valid = validateJson(val);
    
    // Always update parent so inputs don't get out of sync
    handleCustomChange("seo_local_json_ld", val);
  };

  const copyVariable = (text) => {
    navigator.clipboard.writeText(text);
    // Fixed: Added specific ID to prevent toast stacking
    toast.success(`Copied ${text}`, { id: 'copy-variable' });
  };

  const variables = [
    { label: "Site Name", code: "%sitename%", desc: "Global Site Name" },
    { label: "Site URL", code: "%siteurl%", desc: "Frontend URL" },
    { label: "Site Logo", code: "%sitelogo%", desc: "Full Logo URL" },
    { label: "Site Description", code: "%sitedesc%", desc: "Global Description" }
  ];

  return (
    <div className="animate-fade-in max-w-5xl">
      {/* Header */}
      <div className="flex items-start gap-4 p-4 mb-8 rounded-lg bg-indigo-50 dark:bg-indigo-900/20 border border-indigo-100 dark:border-indigo-800/50">
        <div className="p-2 bg-indigo-100 dark:bg-indigo-800 rounded-lg text-indigo-600 dark:text-indigo-300 shrink-0">
          <SafeIcon icon={FiCode} className="w-5 h-5" />
        </div>
        <div>
          <h4 className="font-bold text-gray-900 dark:text-white">Custom Local SEO Schema</h4>
          <p className="text-sm text-gray-600 dark:text-gray-300 mt-1 leading-relaxed">
            Manually configure your Identity schema. Use tools like 
            <a 
              href="https://technicalseo.com/tools/schema-markup-generator/" 
              target="_blank" 
              rel="nofollow noopener noreferrer" 
              className="text-blue-600 hover:underline mx-1 font-medium inline-flex items-center"
            >
              TechnicalSEO <SafeIcon icon={FiExternalLink} className="w-3 h-3 ml-0.5" />
            </a> 
            or 
            <a 
              href="https://www.rankranger.com/schema-markup-generator/" 
              target="_blank" 
              rel="nofollow noopener noreferrer" 
              className="text-blue-600 hover:underline mx-1 font-medium inline-flex items-center"
            >
              RankRanger <SafeIcon icon={FiExternalLink} className="w-3 h-3 ml-0.5" />
            </a>
            to generate the JSON-LD code, then paste it below.
          </p>
        </div>
      </div>

      <div className="grid grid-cols-1 lg:grid-cols-3 gap-8">
        
        {/* Left: Editor */}
        <div className="lg:col-span-2 space-y-4">
            <div className={`relative rounded-xl border transition-colors bg-white dark:bg-gray-800 ${isValid ? 'border-gray-300 dark:border-gray-600 focus-within:border-blue-500' : 'border-red-500 focus-within:border-red-500'}`}>
                <div className="flex justify-between items-center px-4 py-2 border-b border-gray-100 dark:border-gray-700 bg-gray-50 dark:bg-gray-700/50 rounded-t-xl">
                    <span className="text-xs font-bold text-gray-500 dark:text-gray-400 uppercase tracking-wider">JSON-LD Editor</span>
                    {isValid ? (
                        <span className="text-xs text-green-600 dark:text-green-400 flex items-center gap-1">
                            <SafeIcon icon={FiCheckCircle} className="w-3 h-3" /> Valid JSON
                        </span>
                    ) : (
                        <span className="text-xs text-red-600 dark:text-red-400 flex items-center gap-1">
                            <SafeIcon icon={FiAlertTriangle} className="w-3 h-3" /> Invalid Syntax
                        </span>
                    )}
                </div>
                <textarea 
                    value={jsonInput}
                    onChange={handleChange}
                    className="w-full h-96 p-4 font-mono text-sm bg-transparent border-none outline-none resize-y text-gray-800 dark:text-gray-200 placeholder-gray-400 dark:placeholder-gray-500"
                    spellCheck="false"
                    placeholder="{ ... }"
                />
            </div>
            
            {!isValid && (
                <div className="p-3 bg-red-50 dark:bg-red-900/20 text-red-700 dark:text-red-200 text-xs rounded-lg font-mono flex items-start gap-2 border border-red-100 dark:border-red-800">
                    <SafeIcon icon={FiAlertTriangle} className="w-4 h-4 shrink-0 mt-0.5" />
                    <div>
                        <strong>Syntax Error:</strong> {errorMessage}. <br/>
                        Ensure you are pasting <em>only</em> the JSON object (curly braces), not HTML tags.
                    </div>
                </div>
            )}
        </div>

        {/* Right: Variables Helper */}
        <div className="space-y-6">
            <div className="bg-white dark:bg-gray-800 rounded-xl border border-gray-200 dark:border-gray-700 overflow-hidden shadow-sm">
                <div className="p-4 border-b border-gray-100 dark:border-gray-700 bg-gray-50 dark:bg-gray-700/50">
                    <h5 className="font-bold text-gray-900 dark:text-white flex items-center gap-2 text-sm">
                        <SafeIcon icon={FiInfo} className="w-4 h-4 text-blue-500" />
                        Dynamic Variables
                    </h5>
                </div>
                <div className="p-2">
                    {variables.map((v) => (
                        <button
                            key={v.code}
                            onClick={() => copyVariable(v.code)}
                            className="w-full flex items-center justify-between p-3 hover:bg-gray-50 dark:hover:bg-gray-700/50 rounded-lg group transition-colors text-left"
                        >
                            <div>
                                <code className="text-xs font-bold text-blue-600 dark:text-blue-400 bg-blue-50 dark:bg-blue-900/30 px-1.5 py-0.5 rounded">
                                    {v.code}
                                </code>
                                <p className="text-xs text-gray-500 dark:text-gray-400 mt-1">
                                    {v.desc}
                                </p>
                            </div>
                            <SafeIcon icon={FiClipboard} className="w-4 h-4 text-gray-300 dark:text-gray-500 group-hover:text-blue-500 transition-colors" />
                        </button>
                    ))}
                </div>
                <div className="p-4 bg-gray-50 dark:bg-gray-700/30 border-t border-gray-100 dark:border-gray-700 text-xs text-gray-500 dark:text-gray-400 leading-relaxed">
                    Click a variable to copy. These placeholders will be automatically replaced with your actual site data on the frontend.
                </div>
            </div>
        </div>

      </div>
    </div>
  );
};

export default LocalSeoTab;
