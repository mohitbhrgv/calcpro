<?php
// Core utilities
require_once '../../core/config.php';
require_once '../../core/response.php';
require_once '../../core/auth.php';

// --- SECURITY CHECK ---
$admin_data = verifyAdminToken();

try {
    // --- 1. Get Incoming Data ---
    $json = file_get_contents('php://input');
    $data = json_decode($json);

    // --- 2. Validation ---
    if (!isset($data->id) || !is_numeric($data->id)) {
        sendResponse(false, null, 'A valid Media ID is required.', 400, 'INVALID_INPUT');
    }
    // The alt_text key must exist, even if its value is an empty string.
    if (!isset($data->alt_text)) {
        sendResponse(false, null, 'Alt text field is required.', 400, 'INVALID_INPUT');
    }

    // --- 3. Sanitize and Prepare Data ---
    $mediaId = (int)$data->id;
    // Trim the alt text, allowing it to be an empty string if intended.
    $altText = trim($data->alt_text);

    // --- 4. The SQL UPDATE Query ---
    $sql = "UPDATE media SET alt_text = ? WHERE id = ?";
    $stmt = $conn->prepare($sql);
    
    if ($stmt === false) {
        throw new Exception('Failed to prepare the SQL statement.');
    }

    $stmt->bind_param("si", $altText, $mediaId);

    // --- 5. Execute and Send Response ---
    if ($stmt->execute()) {
        // Check if any row was actually changed.
        if ($stmt->affected_rows > 0) {
            sendResponse(true, null, 'Media details updated successfully.', 200);
        } else {
            // If no rows were affected, it could be because the ID doesn't exist,
            // or the new alt text was identical to the old one. We check to be sure.
            $checkSql = "SELECT id FROM media WHERE id = ?";
            $checkStmt = $conn->prepare($checkSql);
            $checkStmt->bind_param("i", $mediaId);
            $checkStmt->execute();
            $checkResult = $checkStmt->get_result();
            if ($checkResult->num_rows === 0) {
                sendResponse(false, null, 'Media item not found.', 404, 'NOT_FOUND');
            } else {
                // The item exists, but no data was changed. This is a success.
                sendResponse(true, null, 'No changes were made to the media details.', 200);
            }
            $checkStmt->close();
        }
    } else {
        throw new Exception("Database update failed: " . $stmt->error);
    }
    $stmt->close();

} catch (Exception $e) {
    error_log("Error in update_media_details.php: " . $e->getMessage());
    sendResponse(false, null, 'An internal server error occurred.', 500, 'SERVER_ERROR');
}

$conn->close();
?>
