
import React, { useState, useEffect, useCallback } from "react";
import { useNavigate } from "react-router-dom";
import { motion, AnimatePresence } from "framer-motion";
import DataTable from "../components/common/DataTable";
import Modal from "../components/common/Modal";
import SafeIcon from "../../common/SafeIcon";
import LoadingSpinner from "../components/common/LoadingSpinner";
import * as FiIcons from "react-icons/fi";
import toast from "react-hot-toast";
import { useAdminCache } from "../context/AdminCacheContext";
import { authenticatedFetch } from "../utils/apiService";

const {
  FiPlus,
  FiShield,
  FiShieldOff,
  FiUser,
  FiMail,
  FiLock,
  FiUsers,
  FiInfo,
  FiTrash2,
  FiEdit2,
  FiCheckCircle,
  FiXCircle,
  FiSearch,
  FiAlertTriangle,
} = FiIcons;

const Users = () => {
  const navigate = useNavigate();
  const { getCache, setCacheData, invalidateCache, isInitialLoadComplete } =
    useAdminCache();

  // --- State ---
  const [currentPage, setCurrentPage] = useState(1);
  const [filterStatus, setFilterStatus] = useState("all");
  const [searchTerm, setSearchTerm] = useState("");

  const [users, setUsers] = useState([]);
  const [totalPages, setTotalPages] = useState(1);
  const [isUpdating, setIsUpdating] = useState(false);
  const [initialLoading, setInitialLoading] = useState(true);

  // Modal States
  const [showDetailsModal, setShowDetailsModal] = useState(false);
  const [selectedUser, setSelectedUser] = useState(null);

  const [showFormModal, setShowFormModal] = useState(false);
  const [isEditing, setIsEditing] = useState(false);
  const [isSubmitting, setIsSubmitting] = useState(false);
  const [formData, setFormData] = useState({
    id: null,
    name: "",
    email: "",
    password: "",
  });

  // Delete Confirmation State
  const [showDeleteModal, setShowDeleteModal] = useState(false);
  const [userToDelete, setUserToDelete] = useState(null);
  const [deleteInput, setDeleteInput] = useState("");

  // Cache Key Generation
  const getCacheKey = (page, status) => `users_page_${page}_status_${status}`;
  const currentKey = getCacheKey(currentPage, filterStatus);

  // --- Fetch Logic ---
  const fetchUsers = useCallback(
    async (forceRefresh = false) => {
      const key = getCacheKey(currentPage, filterStatus);

      if (!forceRefresh) {
        const cached = getCache(key);
        if (cached) {
          setUsers(cached.users);
          setTotalPages(cached.pagination.totalPages);
          setInitialLoading(false);
          return;
        }
      }

      setIsUpdating(true);
      let url = `http://localhost/calcpro-api/api/admin/get_users.php?page=${currentPage}&limit=10`;
      if (filterStatus !== "all") {
        url += `&status=${filterStatus}`;
      }

      try {
        const result = await authenticatedFetch(url);
        if (result.success) {
          setUsers(result.data.users);
          setTotalPages(result.data.pagination.totalPages);
          setCacheData(key, result.data);
        } else {
          toast.error(result.message || "Failed to fetch users.");
        }
      } catch (error) {
        if (error.message !== "Unauthorized")
          toast.error("Could not connect to the server.");
      } finally {
        setIsUpdating(false);
        setInitialLoading(false);
      }
    },
    [currentPage, filterStatus, getCache, setCacheData]
  );

  useEffect(() => {
    fetchUsers(false);
  }, [fetchUsers]);

  // --- Optimistic Ban Toggle ---
  const handleBanToggle = async (user) => {
    const originalStatus = user.status;
    const newStatus = originalStatus === "active" ? "banned" : "active";
    const previousUsers = [...users];

    // Optimistic UI
    if (filterStatus === "all") {
      setUsers((prev) =>
        prev.map((u) => (u.id === user.id ? { ...u, status: newStatus } : u))
      );
    } else {
      setUsers((prev) => prev.filter((u) => u.id !== user.id));
    }

    if (selectedUser?.id === user.id) {
      setSelectedUser((prev) => ({ ...prev, status: newStatus }));
    }

    try {
      const result = await authenticatedFetch(
        "http://localhost/calcpro-api/api/admin/toggle_user_status.php",
        {
          method: "POST",
          body: JSON.stringify({ userId: user.id, newStatus }),
        }
      );

      if (result.success) {
        toast.success(
          `User ${newStatus === "active" ? "activated" : "suspended"} successfully.`
        );
        invalidateCache("users_page_");
      } else {
        throw new Error(result.message);
      }
    } catch (error) {
      toast.error(error.message || "Failed to update status");
      setUsers(previousUsers); // Rollback
      if (selectedUser?.id === user.id)
        setSelectedUser((prev) => ({ ...prev, status: originalStatus }));
    }
  };

  // --- Delete Handling (With Safety Modal) ---
  const initiateDelete = (user) => {
    setUserToDelete(user);
    setDeleteInput("");
    setShowDeleteModal(true);
  };

  const confirmDelete = async () => {
    if (!userToDelete || deleteInput !== "DELETE") return;

    setShowDeleteModal(false);
    const previousUsers = [...users];

    // Optimistic Delete
    setUsers((prev) => prev.filter((u) => u.id !== userToDelete.id));

    try {
      const result = await authenticatedFetch(
        "http://localhost/calcpro-api/api/admin/delete_user.php",
        {
          method: "POST",
          body: JSON.stringify({ userId: userToDelete.id }),
        }
      );
      if (result.success) {
        toast.success("User deleted successfully.");
        invalidateCache("users_page_");
        // Handle empty page case
        if (users.length === 1 && currentPage > 1) {
          setCurrentPage((prev) => prev - 1);
        }
      } else {
        throw new Error(result.message);
      }
    } catch (error) {
      toast.error(error.message || "Delete failed.");
      setUsers(previousUsers); // Rollback
    } finally {
      setUserToDelete(null);
    }
  };

  // --- Form Handlers ---
  const handleOpenAddModal = () => {
    setIsEditing(false);
    setFormData({ id: null, name: "", email: "", password: "" });
    setShowFormModal(true);
  };

  const handleOpenEditModal = (user) => {
    setIsEditing(true);
    setFormData({
      id: user.id,
      name: user.name,
      email: user.email,
      password: "",
    });
    setShowFormModal(true);
  };

  const handleFormChange = (e) => {
    setFormData((prev) => ({ ...prev, [e.target.name]: e.target.value }));
  };

  const handleFormSubmit = async (e) => {
    e.preventDefault();
    setIsSubmitting(true);

    const url = isEditing
      ? "http://localhost/calcpro-api/api/admin/update_user.php"
      : "http://localhost/calcpro-api/api/admin/add_user.php";

    const body = isEditing
      ? { userId: formData.id, name: formData.name, email: formData.email }
      : {
          name: formData.name,
          email: formData.email,
          password: formData.password,
        };

    try {
      const result = await authenticatedFetch(url, {
        method: "POST",
        body: JSON.stringify(body),
      });
      if (result.success) {
        toast.success(result.message);
        setShowFormModal(false);
        invalidateCache("users_page_");
        fetchUsers(true);
      } else {
        toast.error(result.message);
      }
    } catch (error) {
      toast.error("Server connection failed.");
    } finally {
      setIsSubmitting(false);
    }
  };

  // --- Renderers ---
  const columns = [
    {
      key: "name",
      label: "User",
      render: (val, item) => (
        <div className="flex items-center gap-3">
          <div className="w-9 h-9 rounded-full bg-gradient-to-br from-indigo-500 to-violet-600 flex items-center justify-center text-white text-xs font-bold shadow-sm">
            {val.charAt(0).toUpperCase()}
          </div>
          <div className="flex flex-col">
            <span className="text-sm font-semibold text-gray-900 dark:text-white">
              {val}
            </span>
            <span className="text-xs text-gray-400 dark:text-gray-500 font-mono">
              ID: {item.id}
            </span>
          </div>
        </div>
      ),
    },
    {
      key: "email",
      label: "Email",
      render: (val) => (
        <span className="text-sm text-gray-600 dark:text-gray-300">{val}</span>
      ),
    },
    {
      key: "date",
      label: "Joined",
      render: (val) => (
        <span className="text-xs text-gray-500 dark:text-gray-400 font-medium bg-gray-100 dark:bg-gray-700/50 px-2 py-1 rounded">
          {val}
        </span>
      ),
    },
    {
      key: "status",
      label: "Status",
      render: (status) => (
        <span
          className={`inline-flex items-center px-2.5 py-1 rounded-full text-xs font-bold capitalize shadow-sm border ${
            status === "active"
              ? "bg-green-50 text-green-700 border-green-200 dark:bg-green-900/20 dark:text-green-400 dark:border-green-800"
              : "bg-red-50 text-red-700 border-red-200 dark:bg-red-900/20 dark:text-red-400 dark:border-red-800"
          }`}
        >
          {status === "active" ? (
            <FiCheckCircle className="w-3 h-3 mr-1" />
          ) : (
            <FiXCircle className="w-3 h-3 mr-1" />
          )}
          {status}
        </span>
      ),
    },
  ];

  const handleView = (user) => {
    setSelectedUser(user);
    setShowDetailsModal(true);
  };

  const UserDetailsModal = () => (
    <div className="space-y-6">
      <div className="flex items-center gap-5 p-4 bg-gray-50 dark:bg-gray-700/30 rounded-xl border border-gray-100 dark:border-gray-700">
        <div className="w-20 h-20 rounded-full bg-gradient-to-tr from-blue-500 to-indigo-600 flex items-center justify-center text-white text-3xl font-bold shadow-lg">
          {selectedUser?.name.charAt(0).toUpperCase()}
        </div>
        <div className="flex-1">
          <h3 className="text-xl font-bold text-gray-900 dark:text-white">
            {selectedUser?.name}
          </h3>
          <p className="text-sm text-gray-500 dark:text-gray-400 mb-3 font-medium">
            {selectedUser?.email}
          </p>
          <div className="flex gap-2">
            <span
              className={`inline-flex px-3 py-1 rounded-full text-xs font-bold uppercase tracking-wider border ${
                selectedUser?.status === "active"
                  ? "bg-green-50 text-green-700 border-green-200"
                  : "bg-red-50 text-red-700 border-red-200"
              }`}
            >
              {selectedUser?.status}
            </span>
          </div>
        </div>
      </div>

      <div className="flex justify-end pt-2 border-t border-gray-100 dark:border-gray-700">
        <button
          onClick={() => {
            handleBanToggle(selectedUser);
            setShowDetailsModal(false);
          }}
          className={`flex items-center gap-2 px-4 py-2.5 rounded-lg text-sm font-bold shadow-sm transition-all text-white ${
            selectedUser?.status === "active"
              ? "bg-red-600 hover:bg-red-700"
              : "bg-green-600 hover:bg-green-700"
          }`}
        >
          <SafeIcon
            icon={selectedUser?.status === "active" ? FiShieldOff : FiShield}
            className="w-4 h-4"
          />
          <span>
            {selectedUser?.status === "active"
              ? "Suspend Account"
              : "Activate Account"}
          </span>
        </button>
      </div>
    </div>
  );

  if (!isInitialLoadComplete || initialLoading)
    return (
      <div className="h-96 flex items-center justify-center">
        <LoadingSpinner size="xl" />
      </div>
    );

  return (
    <div className="space-y-6">
      {/* --- Standardized Header Card --- */}
      <div className="flex items-start gap-4 p-4 rounded-lg bg-indigo-50 dark:bg-indigo-900/20 border border-indigo-100 dark:border-indigo-800/50">
        <div className="p-2 bg-indigo-100 dark:bg-indigo-800 rounded-lg text-indigo-600 dark:text-indigo-300 shrink-0">
          <SafeIcon icon={FiUsers} className="w-5 h-5" />
        </div>
        <div className="flex-1">
          <div className="flex flex-col sm:flex-row sm:items-center justify-between gap-4">
            <div>
              <h4 className="font-bold text-gray-900 dark:text-white">
                User Management
              </h4>
              <p className="text-sm text-gray-600 dark:text-gray-300 mt-1 leading-relaxed">
                Manage registered users, monitor account status, and control
                access permissions.
              </p>
            </div>

            {/* Main Action Button */}
            <motion.button
              whileHover={{ scale: 1.02 }}
              whileTap={{ scale: 0.98 }}
              onClick={handleOpenAddModal}
              className="flex items-center justify-center space-x-2 bg-indigo-600 text-white px-5 py-2.5 rounded-lg shadow-sm hover:bg-indigo-700 transition-colors font-medium text-sm whitespace-nowrap"
            >
              <SafeIcon icon={FiPlus} className="w-4 h-4" />
              <span>Add User</span>
            </motion.button>
          </div>
        </div>
      </div>

      {/* Delete Confirmation Modal */}
      <Modal
        isOpen={showDeleteModal}
        onClose={() => setShowDeleteModal(false)}
        title="Confirm Deletion"
        type="danger"
        size="sm"
        disableAnimation={true} // --- INSTANT POPUP ---
      >
        <div className="space-y-4">
          <div className="bg-red-50 dark:bg-red-900/20 p-4 rounded-lg border border-red-200 dark:border-red-800 flex items-start gap-3">
            <SafeIcon
              icon={FiAlertTriangle}
              className="w-6 h-6 text-red-600 shrink-0 mt-0.5"
            />
            <div>
              <h4 className="text-sm font-bold text-red-900 dark:text-red-100">
                Permanent Action
              </h4>
              <p className="text-xs text-red-800 dark:text-red-200/80 mt-1">
                You are about to delete user{" "}
                <strong>{userToDelete?.name}</strong>. This will erase all their
                data including calculations and bookmarks.
              </p>
            </div>
          </div>
          <div>
            <label className="block text-xs font-bold text-gray-600 dark:text-gray-400 mb-2 uppercase">
              Type "DELETE" to confirm
            </label>
            <input
              type="text"
              value={deleteInput}
              onChange={(e) => setDeleteInput(e.target.value)}
              className="w-full px-3 py-2 border border-gray-300 dark:border-gray-600 rounded-lg focus:ring-2 focus:ring-red-500 dark:bg-gray-700 dark:text-white outline-none"
              placeholder="DELETE"
            />
          </div>
          <div className="flex justify-end gap-3 pt-2">
            <button
              onClick={() => setShowDeleteModal(false)}
              className="px-4 py-2 text-sm font-medium text-gray-700 hover:bg-gray-100 dark:text-gray-300 dark:hover:bg-gray-700 rounded-lg transition-colors"
            >
              Cancel
            </button>
            <button
              onClick={confirmDelete}
              disabled={deleteInput !== "DELETE"}
              className="px-4 py-2 text-sm font-bold text-white bg-red-600 hover:bg-red-700 rounded-lg shadow-sm transition-colors disabled:opacity-50 disabled:cursor-not-allowed"
            >
              Delete User
            </button>
          </div>
        </div>
      </Modal>

      <div className="bg-white dark:bg-gray-800 rounded-xl shadow-sm border border-gray-200 dark:border-gray-700 overflow-hidden relative min-h-[400px]">
        {/* Loading Overlay */}
        <AnimatePresence>
          {isUpdating && (
            <motion.div
              initial={{ opacity: 0 }}
              animate={{ opacity: 1 }}
              exit={{ opacity: 0 }}
              className="absolute inset-0 bg-white/60 dark:bg-gray-900/60 backdrop-blur-[1px] z-10 flex items-center justify-center"
            >
              <LoadingSpinner size="lg" />
            </motion.div>
          )}
        </AnimatePresence>

        <DataTable
          columns={columns}
          data={users}
          loading={false}
          onView={handleView}
          onEdit={handleOpenEditModal}
          onDelete={initiateDelete} // Uses the new safe delete initator
          currentPage={currentPage}
          totalPages={totalPages}
          onPageChange={setCurrentPage}
          searchable={false} // Disable default search to use custom one
          headerLeftControls={
            <div className="flex items-center gap-1 bg-gray-100 dark:bg-gray-700/50 p-1 rounded-lg">
              {[
                { id: "all", label: "All Users" },
                { id: "active", label: "Active" },
                { id: "banned", label: "Banned" },
              ].map((tab) => (
                <button
                  key={tab.id}
                  onClick={() => {
                    setFilterStatus(tab.id);
                    setSearchTerm("");
                    setCurrentPage(1);
                  }}
                  className={`px-4 py-1.5 text-xs font-bold rounded-md transition-all ${
                    filterStatus === tab.id
                      ? "bg-white dark:bg-gray-800 text-indigo-600 dark:text-indigo-400 shadow-sm ring-1 ring-black/5"
                      : "text-gray-500 hover:text-gray-700 dark:text-gray-400 dark:hover:text-gray-200"
                  }`}
                >
                  {tab.label}
                </button>
              ))}
            </div>
          }
          headerRightControls={
            // Custom Search Input placed on the Right
            <div className="relative">
              <SafeIcon
                icon={FiSearch}
                className="absolute left-3 top-1/2 -translate-y-1/2 text-gray-400 w-4 h-4"
              />
              <input
                type="text"
                placeholder="Search users..."
                value={searchTerm}
                onChange={(e) => setSearchTerm(e.target.value)}
                className="pl-9 pr-4 py-1.5 border border-gray-300 dark:border-gray-600 rounded-lg text-sm focus:ring-2 focus:ring-indigo-500 dark:bg-gray-700 dark:text-white w-64 transition-shadow outline-none"
              />
            </div>
          }
        />
      </div>

      {/* View Details Modal */}
      <Modal
        isOpen={showDetailsModal}
        onClose={() => setShowDetailsModal(false)}
        title="User Details"
        size="md"
        disableAnimation={true} // --- INSTANT POPUP ---
      >
        {selectedUser && <UserDetailsModal />}
      </Modal>

      {/* Add/Edit Modal */}
      <Modal
        isOpen={showFormModal}
        onClose={() => setShowFormModal(false)}
        title={isEditing ? "Edit User" : "Add New User"}
        size="md"
        disableAnimation={true} // --- INSTANT POPUP ---
      >
        <form onSubmit={handleFormSubmit} className="space-y-5">
          <div>
            <label className="block text-xs font-bold text-gray-500 dark:text-gray-400 uppercase mb-1.5">
              Full Name
            </label>
            <div className="relative">
              <div className="absolute inset-y-0 left-0 pl-3 flex items-center pointer-events-none text-gray-400">
                <SafeIcon icon={FiUser} className="w-5 h-5" />
              </div>
              <input
                type="text"
                name="name"
                value={formData.name}
                onChange={handleFormChange}
                required
                className="w-full pl-10 pr-4 py-2.5 border border-gray-300 dark:border-gray-600 rounded-lg dark:bg-gray-700 dark:text-white focus:ring-2 focus:ring-indigo-500 outline-none transition-shadow"
                placeholder="John Doe"
              />
            </div>
          </div>
          <div>
            <label className="block text-xs font-bold text-gray-500 dark:text-gray-400 uppercase mb-1.5">
              Email Address
            </label>
            <div className="relative">
              <div className="absolute inset-y-0 left-0 pl-3 flex items-center pointer-events-none text-gray-400">
                <SafeIcon icon={FiMail} className="w-5 h-5" />
              </div>
              <input
                type="email"
                name="email"
                value={formData.email}
                onChange={handleFormChange}
                required
                className="w-full pl-10 pr-4 py-2.5 border border-gray-300 dark:border-gray-600 rounded-lg dark:bg-gray-700 dark:text-white focus:ring-2 focus:ring-indigo-500 outline-none transition-shadow"
                placeholder="john@example.com"
              />
            </div>
          </div>
          {!isEditing && (
            <div>
              <label className="block text-xs font-bold text-gray-500 dark:text-gray-400 uppercase mb-1.5">
                Password
              </label>
              <div className="relative">
                <div className="absolute inset-y-0 left-0 pl-3 flex items-center pointer-events-none text-gray-400">
                  <SafeIcon icon={FiLock} className="w-5 h-5" />
                </div>
                <input
                  type="password"
                  name="password"
                  value={formData.password}
                  onChange={handleFormChange}
                  required={!isEditing}
                  className="w-full pl-10 pr-4 py-2.5 border border-gray-300 dark:border-gray-600 rounded-lg dark:bg-gray-700 dark:text-white focus:ring-2 focus:ring-indigo-500 outline-none transition-shadow"
                  placeholder="••••••••"
                />
              </div>
              <p className="text-[10px] text-gray-400 mt-1 flex items-center gap-1">
                <SafeIcon icon={FiInfo} className="w-3 h-3" />
                Must be 8+ characters with a letter, number, & symbol.
              </p>
            </div>
          )}
          <div className="flex justify-end gap-3 pt-4 border-t border-gray-100 dark:border-gray-700">
            <button
              type="button"
              onClick={() => setShowFormModal(false)}
              className="px-4 py-2 text-sm font-medium text-gray-600 dark:text-gray-300 hover:bg-gray-100 dark:hover:bg-gray-700 rounded-lg transition-colors"
            >
              Cancel
            </button>
            <button
              type="submit"
              disabled={isSubmitting}
              className="flex items-center gap-2 px-6 py-2 bg-indigo-600 hover:bg-indigo-700 text-white text-sm font-bold rounded-lg shadow-sm transition-all disabled:opacity-50"
            >
              {isSubmitting ? (
                <LoadingSpinner size="xs" color="white" />
              ) : (
                <SafeIcon
                  icon={isEditing ? FiEdit2 : FiPlus}
                  className="w-4 h-4"
                />
              )}
              {isSubmitting
                ? "Saving..."
                : isEditing
                  ? "Update User"
                  : "Create User"}
            </button>
          </div>
        </form>
      </Modal>
    </div>
  );
};

export default Users;
