import React, { useState } from "react";
import { motion, AnimatePresence } from "framer-motion";
import SafeIcon from "@/common/SafeIcon";
import * as FiIcons from "react-icons/fi";

const {
  FiX, FiSearch, FiFileText, FiBook, FiMonitor, FiCalendar,
  FiHelpCircle, FiCheckSquare, FiTool, FiBriefcase, FiMusic,
  FiUser, FiMic, FiShoppingBag, FiCoffee, FiMapPin, FiDisc
} = FiIcons;

const SchemaCatalogModal = ({ isOpen, onClose, onSelect }) => {
  const [searchQuery, setSearchQuery] = useState("");

  const schemaTypes = [
    { id: 'Article', label: 'Article', icon: FiFileText, desc: 'News, Blog Posts, Reports' },
    { id: 'Book', label: 'Book', icon: FiBook, desc: 'Books and Authors' },
    { id: 'Course', label: 'Course', icon: FiMonitor, desc: 'Educational Courses' },
    { id: 'Event', label: 'Event', icon: FiCalendar, desc: 'Concerts, Festivals' },
    { id: 'FAQPage', label: 'FAQ', icon: FiHelpCircle, desc: 'Frequently Asked Questions' },
    { id: 'FactCheck', label: 'Fact Check', icon: FiCheckSquare, desc: 'Fact Checking Content', isPro: true },
    { id: 'HowTo', label: 'HowTo', icon: FiTool, desc: 'Step-by-step Instructions', isPro: true },
    { id: 'JobPosting', label: 'Job Posting', icon: FiBriefcase, desc: 'Job Vacancies' },
    { id: 'Music', label: 'Music', icon: FiMusic, desc: 'Music Albums/Playlists' },
    { id: 'Person', label: 'Person', icon: FiUser, desc: 'Personal Profiles' },
    { id: 'PodcastEpisode', label: 'Podcast', icon: FiMic, desc: 'Podcast Episodes', isPro: true },
    { id: 'Product', label: 'Product', icon: FiShoppingBag, desc: 'Ecommerce Products' },
    { id: 'Recipe', label: 'Recipe', icon: FiCoffee, desc: 'Cooking Recipes' },
    { id: 'Restaurant', label: 'Restaurant', icon: FiMapPin, desc: 'Food Establishments' },
    { id: 'Service', label: 'Service', icon: FiBriefcase, desc: 'Professional Services' },
    { id: 'SoftwareApplication', label: 'Software', icon: FiDisc, desc: 'Apps and Software' },
  ];

  const filteredSchemas = schemaTypes.filter(s => 
    s.label.toLowerCase().includes(searchQuery.toLowerCase())
  );

  return (
    <AnimatePresence>
      {isOpen && (
        <div className="fixed inset-0 z-[100] flex items-center justify-center p-4 sm:p-6">
          <motion.div 
            initial={{ opacity: 0 }} 
            animate={{ opacity: 1 }} 
            exit={{ opacity: 0 }} 
            onClick={onClose}
            className="fixed inset-0 bg-gray-900/60 backdrop-blur-sm"
          />

          <motion.div 
            initial={{ scale: 0.95, opacity: 0, y: 20 }} 
            animate={{ scale: 1, opacity: 1, y: 0 }} 
            exit={{ scale: 0.95, opacity: 0, y: 20 }} 
            className="bg-white dark:bg-gray-800 w-full max-w-4xl rounded-xl shadow-2xl flex flex-col max-h-[85vh] relative z-10 overflow-hidden font-sans"
          >
            {/* Header */}
            <div className="px-8 py-6 border-b border-gray-100 dark:border-gray-700 flex justify-between items-center">
              <div>
                <h3 className="text-xl font-bold text-gray-900 dark:text-white">Schema Generator</h3>
                <p className="text-sm text-gray-500 dark:text-gray-400 mt-1">
                  Select a schema type to add to your post.
                </p>
              </div>
              <div className="flex items-center gap-4">
                <div className="relative">
                    <SafeIcon icon={FiSearch} className="absolute left-3 top-1/2 -translate-y-1/2 w-4 h-4 text-gray-400" />
                    <input 
                        type="text" 
                        placeholder="Search Schema..." 
                        value={searchQuery}
                        onChange={(e) => setSearchQuery(e.target.value)}
                        className="pl-9 pr-4 py-2 bg-gray-100 dark:bg-gray-700 border-transparent focus:border-blue-500 focus:bg-white dark:focus:bg-gray-600 rounded-lg text-sm text-gray-900 dark:text-white transition-all w-64 focus:ring-0"
                    />
                </div>
                <button onClick={onClose} className="text-gray-400 hover:text-gray-600 dark:hover:text-gray-200 transition-colors">
                    <SafeIcon icon={FiX} className="w-6 h-6" />
                </button>
              </div>
            </div>

            {/* Grid Content */}
            <div className="flex-1 overflow-y-auto p-8 bg-gray-50 dark:bg-gray-900/50">
                <div className="grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-3 gap-4">
                    {filteredSchemas.map((schema) => (
                        <div 
                            key={schema.id} 
                            className="bg-white dark:bg-gray-800 border border-gray-200 dark:border-gray-700 rounded-xl p-5 hover:shadow-md hover:border-blue-300 dark:hover:border-blue-700 transition-all group relative"
                        >
                            {schema.isPro && (
                                <span className="absolute top-3 right-3 bg-gradient-to-r from-amber-500 to-orange-500 text-white text-[10px] font-bold px-2 py-0.5 rounded-full uppercase tracking-wide shadow-sm">
                                    PRO
                                </span>
                            )}
                            
                            <div className="flex items-start justify-between mb-3">
                                <div className="w-12 h-12 rounded-lg bg-blue-50 dark:bg-blue-900/20 flex items-center justify-center text-blue-600 dark:text-blue-400 group-hover:scale-110 transition-transform">
                                    <SafeIcon icon={schema.icon} className="w-6 h-6" />
                                </div>
                                <button 
                                    onClick={() => onSelect(schema.id)}
                                    className="px-4 py-1.5 rounded-md bg-gray-100 dark:bg-gray-700 text-gray-700 dark:text-gray-300 text-xs font-bold uppercase tracking-wide hover:bg-blue-600 hover:text-white dark:hover:bg-blue-600 transition-colors"
                                >
                                    Use
                                </button>
                            </div>
                            
                            <h4 className="text-base font-bold text-gray-900 dark:text-white mb-1">
                                {schema.label}
                            </h4>
                            <p className="text-sm text-gray-500 dark:text-gray-400 line-clamp-2">
                                {schema.desc}
                            </p>
                        </div>
                    ))}
                </div>
            </div>

          </motion.div>
        </div>
      )}
    </AnimatePresence>
  );
};

export default SchemaCatalogModal;
