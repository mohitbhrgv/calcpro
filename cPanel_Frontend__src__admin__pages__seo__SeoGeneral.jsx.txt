import React, { useState, useEffect, useMemo } from "react";
import toast from "react-hot-toast";
import SafeIcon from "../../../common/SafeIcon";
import LoadingSpinner from "../../components/common/LoadingSpinner";
import Modal from "../../components/common/Modal";
import { authenticatedFetch } from "../../utils/apiService";
import * as FiIcons from "react-icons/fi";
import { motion } from "framer-motion";
import BreadcrumbsTab from "../../components/seo/tabs/BreadcrumbsTab";
import BreadcrumbHelpModal from "../../components/help/BreadcrumbHelpModal";
import { useAdminCache } from "../../context/AdminCacheContext";

const { FiSave, FiList, FiInfo, FiAlertTriangle } = FiIcons;

const SeoGeneral = () => {
  const { getCache, setCacheData, isInitialLoadComplete } = useAdminCache();
  const cacheKey = "settings_general";
  const cachedData = getCache(cacheKey);

  const [activeTab, setActiveTab] = useState("breadcrumbs");
  const [loading, setLoading] = useState(!cachedData);
  const [isSubmitting, setIsSubmitting] = useState(false);
  
  const [isHelpOpen, setIsHelpOpen] = useState(false);
  const [showConfirmModal, setShowConfirmModal] = useState(false);
  const [confirmInput, setConfirmInput] = useState("");

  const parseSettings = (data) => ({
    seo_breadcrumbs_enable: data.seo_breadcrumbs_enable ?? true,
    
    // Text Fields
    seo_breadcrumbs_separator: data.seo_breadcrumbs_separator || '', 
    seo_breadcrumbs_home_label: data.seo_breadcrumbs_home_label || '',
    seo_breadcrumbs_archive_format: data.seo_breadcrumbs_archive_format || '',
    seo_breadcrumbs_search_format: data.seo_breadcrumbs_search_format || '',
    seo_breadcrumbs_404_label: data.seo_breadcrumbs_404_label || '',
    seo_breadcrumbs_prefix: data.seo_breadcrumbs_prefix || '',
    
    seo_breadcrumbs_show_home: data.seo_breadcrumbs_show_home ?? true,
    seo_breadcrumbs_show_category: data.seo_breadcrumbs_show_category ?? true,
    
    // Visibility Toggles
    seo_breadcrumbs_show_pages: data.seo_breadcrumbs_show_pages ?? true,
    seo_breadcrumbs_show_posts: data.seo_breadcrumbs_show_posts ?? true,
    seo_breadcrumbs_show_blog_categories: data.seo_breadcrumbs_show_blog_categories ?? true,
    seo_breadcrumbs_show_calculators: data.seo_breadcrumbs_show_calculators ?? true,
    seo_breadcrumbs_show_calc_categories: data.seo_breadcrumbs_show_calc_categories ?? true,
    seo_breadcrumbs_show_prefix: data.seo_breadcrumbs_show_prefix ?? true,
    seo_breadcrumbs_archive_format_enable: data.seo_breadcrumbs_archive_format_enable ?? true,

    // Advanced
    seo_breadcrumbs_link_category: data.seo_breadcrumbs_link_category ?? false,
    seo_breadcrumbs_hide_post_title: data.seo_breadcrumbs_hide_post_title ?? false,
  });

  const [settings, setSettings] = useState(
    cachedData ? { ...cachedData, ...parseSettings(cachedData) } : {}
  );
  const [initialSettings, setInitialSettings] = useState(
    cachedData ? { ...cachedData, ...parseSettings(cachedData) } : {}
  );

  const hasChanges = useMemo(() => {
    return JSON.stringify(settings) !== JSON.stringify(initialSettings);
  }, [settings, initialSettings]);

  useEffect(() => {
    const loadData = async () => {
      if (cachedData) {
        setLoading(false);
        return;
      }
      try {
        const result = await authenticatedFetch("http://localhost/calcpro-api/api/admin/get_settings.php");
        if (result.success) {
          const parsed = parseSettings(result.data);
          const fullSettings = { ...result.data, ...parsed };
          setSettings(fullSettings);
          setInitialSettings(fullSettings);
          setCacheData(cacheKey, fullSettings);
        }
      } catch (e) {
        toast.error("Failed to load settings.");
      } finally {
        setLoading(false);
      }
    };
    if (!cachedData) {
        loadData();
    }
  }, [cachedData, cacheKey, setCacheData]);

  const handleCustomChange = (name, value) => {
    setSettings(prev => ({ ...prev, [name]: value }));
  };
  
  const handleSaveClick = () => {
    if (!hasChanges) return;
    setConfirmInput("");
    setShowConfirmModal(true);
  };

  const executeSave = async () => {
    setShowConfirmModal(false);
    setIsSubmitting(true);
    const minDelay = new Promise(resolve => setTimeout(resolve, 800));

    try {
      const apiCall = authenticatedFetch("http://localhost/calcpro-api/api/admin/update_settings.php", {
        method: "POST",
        body: JSON.stringify(settings),
      });

      const [result] = await Promise.all([apiCall, minDelay]);

      if(result.success) {
        toast.success("Settings saved successfully!");
        const newInitialState = { ...settings };
        setInitialSettings(newInitialState);
        setCacheData(cacheKey, newInitialState);
      } else {
        toast.error(result.message || "Failed to save.");
      }
    } catch (e) {
      toast.error("Connection failed.");
    } finally {
      setIsSubmitting(false);
    }
  };

  const renderTabContent = () => {
    const commonProps = { 
        settings, 
        handleCustomChange,
        onSave: handleSaveClick, 
        hasChanges,
        isSubmitting
    };
    switch (activeTab) {
      case 'breadcrumbs':
        return <BreadcrumbsTab {...commonProps} />;
      default:
        return <div>Select a tab</div>;
    }
  };

  if (!isInitialLoadComplete && loading) return <LoadingSpinner size="xl" className="h-64" />;

  return (
    <div className="flex flex-col md:flex-row bg-gray-50 dark:bg-gray-900 font-sans text-gray-900 dark:text-gray-100 min-h-full">
       
       {/* --- 1. HELP MODAL (No Animation) --- */}
       <BreadcrumbHelpModal 
          isOpen={isHelpOpen} 
          onClose={() => setIsHelpOpen(false)} 
          disableAnimation={true}
       />
       
       {/* --- 2. CONFIRM MODAL (No Animation) --- */}
       <Modal
          isOpen={showConfirmModal}
          onClose={() => setShowConfirmModal(false)}
          title="Confirm General SEO Changes"
          type="warning"
          size="md"
          disableAnimation={true} // --- INSTANT POPUP ---
        >
          <div className="space-y-4">
              <div className="bg-amber-50 dark:bg-amber-900/20 p-4 rounded-lg border border-amber-200 dark:border-amber-800 flex items-start gap-3">
                  <SafeIcon icon={FiAlertTriangle} className="w-6 h-6 text-amber-600 shrink-0 mt-0.5" />
                  <div>
                      <h4 className="text-sm font-bold text-amber-900 dark:text-amber-100">Confirm Changes</h4>
                      <p className="text-xs text-amber-800 dark:text-amber-200/80 mt-1">
                          You are about to modify site-wide SEO settings. Please confirm this action.
                      </p>
                  </div>
              </div>
              <div>
                  <label className="block text-sm font-medium text-gray-700 dark:text-gray-300 mb-2">
                      Type <strong>CONFIRM</strong> to proceed:
                  </label>
                  <input
                      type="text"
                      value={confirmInput}
                      onChange={(e) => setConfirmInput(e.target.value)}
                      className="w-full px-3 py-2 border border-gray-300 dark:border-gray-600 rounded-md focus:ring-2 focus:ring-amber-500 dark:bg-gray-700 dark:text-white"
                      placeholder="CONFIRM"
                  />
              </div>
              <div className="flex justify-end gap-3 pt-2">
                  <button onClick={() => setShowConfirmModal(false)} className="px-4 py-2 text-gray-700 dark:text-gray-300 hover:bg-gray-100 dark:hover:bg-gray-700 rounded-lg transition-colors">
                      Cancel
                  </button>
                  <button
                      onClick={executeSave}
                      disabled={confirmInput !== "CONFIRM"}
                      className="px-4 py-2 bg-amber-600 text-white rounded-lg hover:bg-amber-700 disabled:opacity-50 disabled:cursor-not-allowed transition-colors font-medium shadow-sm"
                  >
                      Save Changes
                  </button>
              </div>
          </div>
        </Modal>

      {/* Internal Sidebar */}
      <div className="w-full md:w-64 bg-white dark:bg-gray-800 border-b md:border-b-0 md:border-r border-gray-200 dark:border-gray-700 flex-shrink-0">
         <div className="p-5 border-b border-gray-200 dark:border-gray-700">
            <h2 className="font-bold text-lg text-gray-900 dark:text-white">General SEO</h2>
         </div>
         <nav className="py-2">
            <button
                onClick={() => setActiveTab('breadcrumbs')}
                className={`w-full flex items-center px-5 py-3 text-sm font-medium transition-colors border-l-4 ${
                    activeTab === 'breadcrumbs'
                    ? "border-blue-500 bg-blue-50 dark:bg-blue-900/20 text-blue-600 dark:text-blue-400"
                    : "border-transparent text-gray-600 dark:text-gray-400 hover:bg-gray-50 dark:hover:bg-gray-700"
                }`}
            >
                <SafeIcon icon={FiList} className={`w-4 h-4 mr-3 ${activeTab === 'breadcrumbs' ? "text-blue-500" : "text-gray-400"}`} />
                Breadcrumbs
            </button>
         </nav>
      </div>

      {/* Content Area */}
      <div className="flex-1 p-4 md:p-8 overflow-y-auto">
         <div className="max-w-5xl mx-auto">
            
            <div className="flex items-center justify-between mb-8">
                <div className="flex items-center gap-2">
                    <h1 className="text-3xl font-bold text-gray-900 dark:text-white">
                        Breadcrumbs
                    </h1>
                    <button
                      onClick={() => setIsHelpOpen(true)}
                      className="text-gray-400 hover:text-indigo-600 dark:hover:text-indigo-400 transition-colors p-1 rounded-full hover:bg-indigo-50 dark:hover:bg-indigo-900/30"
                      title="Configuration Guide"
                    >
                      <SafeIcon icon={FiInfo} className="w-5 h-5" />
                    </button>
                </div>
                 <motion.button
                  whileHover={hasChanges ? { scale: 1.05 } : {}}
                  whileTap={hasChanges ? { scale: 0.95 } : {}}
                  onClick={handleSaveClick}
                  disabled={isSubmitting || !hasChanges}
                  className={`flex items-center space-x-2 px-5 py-2.5 rounded-lg transition-colors shadow-sm font-medium ${
                    hasChanges
                      ? "bg-calcpro-primary text-white hover:bg-calcpro-primary/90"
                      : "bg-gray-200 dark:bg-gray-700 text-gray-500 dark:text-gray-400 cursor-not-allowed"
                  }`}
                >
                  <SafeIcon icon={FiSave} className="w-4 h-4" />
                  <span>{isSubmitting ? 'Saving...' : 'Save Changes'}</span>
                </motion.button>
            </div>

            <div className="bg-white dark:bg-gray-800 rounded-xl shadow-sm border border-gray-200 dark:border-gray-700 p-8">
                {renderTabContent()}
            </div>
         </div>
      </div>
    </div>
  );
};

export default SeoGeneral;
