import React, { useState, useEffect } from "react";
import { useParams, Link } from "react-router-dom";
import { motion } from "framer-motion";
import toast from "react-hot-toast";
// --- THIS IS THE CORRECTED IMPORT PATH ---
import SafeIcon from "../../common/SafeIcon";
import { useLayout } from "../context/LayoutContext";
import {
  FiMail,
  FiCheckCircle,
  FiLock,
  FiSave,
  FiArrowLeft,
} from "react-icons/fi";

const ResetPassword = () => {
  const { token } = useParams();
  const { setBackgroundClass } = useLayout();

  const [uiState, setUiState] = useState(token ? "resetForm" : "requestForm");
  const [email, setEmail] = useState("");
  const [password, setPassword] = useState("");
  const [confirmPassword, setConfirmPassword] = useState("");
  const [isSubmitting, setIsSubmitting] = useState(false);
  const [error, setError] = useState("");

  // Set the unique background for the reset page
  useEffect(() => {
    setBackgroundClass(
      "bg-gradient-to-br from-indigo-600 via-blue-600 to-blue-500"
    );
    // Cleanup function to reset the background when the component unmounts
    return () => {
      setBackgroundClass("bg-gray-50 dark:bg-gray-900");
    };
  }, [setBackgroundClass]);

  const handleRequestSubmit = async (e) => {
    e.preventDefault();
    setError("");
    if (!email) {
      setError("Please enter your email address.");
      return;
    }
    setIsSubmitting(true);
    const toastId = "reset-toast";

    try {
      const response = await fetch(
        "http://localhost/calcpro-api/api/admin/request_admin_password_reset.php",
        {
          method: "POST",
          headers: { "Content-Type": "application/json" },
          body: JSON.stringify({ email }),
        }
      );
      const result = await response.json();

      // We always show a success message to prevent email enumeration attacks
      toast.success(result.message, { id: toastId });
      setUiState("emailSent");
    } catch (error) {
      toast.error("Could not connect to the server. Please try again.", {
        id: toastId,
      });
    } finally {
      setIsSubmitting(false);
    }
  };

  const handleResetSubmit = async (e) => {
    e.preventDefault();
    setError("");
    if (password !== confirmPassword) {
      setError("Passwords do not match.");
      return;
    }
    if (password.length < 8) {
      setError("Password must be at least 8 characters long.");
      return;
    }

    setIsSubmitting(true);
    const toastId = "reset-toast";

    try {
      const response = await fetch(
        "http://localhost/calcpro-api/api/admin/reset_admin_password.php",
        {
          method: "POST",
          headers: { "Content-Type": "application/json" },
          body: JSON.stringify({ token, password }),
        }
      );

      const result = await response.json();

      if (result.success) {
        toast.success(result.message, { id: toastId });
        setUiState("passwordReset");
      } else {
        toast.error(result.message || "An unknown error occurred.", {
          id: toastId,
        });
      }
    } catch (error) {
      toast.error("Could not connect to the server.", { id: toastId });
    } finally {
      setIsSubmitting(false);
    }
  };

  const renderContent = () => {
    switch (uiState) {
      case "requestForm":
        return (
          <form onSubmit={handleRequestSubmit} className="space-y-6">
            <div className="text-center">
              <h1 className="text-2xl font-bold text-neutral-900 dark:text-white">
                Forgot Password?
              </h1>
              <p className="text-neutral-600 dark:text-neutral-400 mt-2">
                Enter your email to receive reset instructions.
              </p>
            </div>
            <div>
              <label
                htmlFor="email"
                className="block text-sm font-medium text-neutral-700 dark:text-neutral-300 mb-2"
              >
                Email Address
              </label>
              <div className="relative">
                <SafeIcon
                  icon={FiMail}
                  className="absolute left-3 top-1/2 -translate-y-1/2 h-5 w-5 text-neutral-400"
                />
                <input
                  id="email"
                  type="email"
                  value={email}
                  onChange={(e) => setEmail(e.target.value)}
                  className="w-full pl-10 pr-4 py-3 border border-neutral-300 dark:border-neutral-600 rounded-lg focus:ring-2 focus:ring-primary-500 bg-white/50 dark:bg-neutral-700/50"
                />
              </div>
              {error && <p className="mt-1 text-sm text-red-500">{error}</p>}
            </div>
            <button
              type="submit"
              disabled={isSubmitting}
              className="w-full flex items-center justify-center space-x-2 bg-gradient-to-r from-primary-600 to-indigo-600 text-white px-6 py-3 rounded-lg font-medium"
            >
              {isSubmitting ? (
                <div className="w-5 h-5 border-2 border-white border-t-transparent rounded-full animate-spin"></div>
              ) : (
                <span>Send Instructions</span>
              )}
            </button>
          </form>
        );
      case "resetForm":
        return (
          <form onSubmit={handleResetSubmit} className="space-y-6">
            <div className="text-center">
              <h1 className="text-2xl font-bold text-neutral-900 dark:text-white">
                Set New Password
              </h1>
              <p className="text-neutral-600 dark:text-neutral-400 mt-2">
                Please enter and confirm your new password.
              </p>
            </div>
            <div>
              <label className="block text-sm font-medium text-neutral-700 dark:text-neutral-300 mb-2">
                New Password
              </label>
              <div className="relative">
                <SafeIcon
                  icon={FiLock}
                  className="absolute left-3 top-1/2 -translate-y-1/2 h-5 w-5 text-neutral-400"
                />
                <input
                  type="password"
                  value={password}
                  onChange={(e) => setPassword(e.target.value)}
                  className="w-full pl-10 pr-4 py-3 border border-neutral-300 dark:border-neutral-600 rounded-lg"
                />
              </div>
            </div>
            <div>
              <label className="block text-sm font-medium text-neutral-700 dark:text-neutral-300 mb-2">
                Confirm New Password
              </label>
              <div className="relative">
                <SafeIcon
                  icon={FiLock}
                  className="absolute left-3 top-1/2 -translate-y-1/2 h-5 w-5 text-neutral-400"
                />
                <input
                  type="password"
                  value={confirmPassword}
                  onChange={(e) => setConfirmPassword(e.target.value)}
                  className="w-full pl-10 pr-4 py-3 border border-neutral-300 dark:border-neutral-600 rounded-lg"
                />
              </div>
            </div>
            {error && <p className="mt-1 text-sm text-red-500">{error}</p>}
            <button
              type="submit"
              disabled={isSubmitting}
              className="w-full flex items-center justify-center space-x-2 bg-gradient-to-r from-primary-600 to-indigo-600 text-white px-6 py-3 rounded-lg font-medium"
            >
              {isSubmitting ? (
                <div className="w-5 h-5 border-2 border-white border-t-transparent rounded-full animate-spin"></div>
              ) : (
                <>
                  <SafeIcon icon={FiSave} />
                  <span>Reset Password</span>
                </>
              )}
            </button>
          </form>
        );
      case "emailSent":
      case "passwordReset":
        return (
          <div className="text-center">
            <div className="w-16 h-16 bg-green-100 dark:bg-green-900/30 rounded-full flex items-center justify-center mx-auto mb-4">
              <SafeIcon
                icon={FiCheckCircle}
                className="w-8 h-8 text-green-500"
              />
            </div>
            <h2 className="text-2xl font-bold text-neutral-900 dark:text-white mb-2">
              {uiState === "emailSent"
                ? "Check Your Email"
                : "Password Updated!"}
            </h2>
            <p className="text-neutral-600 dark:text-neutral-400">
              {uiState === "emailSent"
                ? `We've sent password reset instructions to ${email}.`
                : "Your password has been reset successfully."}
            </p>
          </div>
        );
    }
  };

  return (
    <div className="min-h-screen flex items-center justify-center px-4">
      <motion.div
        initial={{ opacity: 0, y: 20 }}
        animate={{ opacity: 1, y: 0 }}
        transition={{ duration: 0.5 }}
        className="w-full max-w-md bg-white/80 dark:bg-neutral-800/80 backdrop-blur-md rounded-xl shadow-2xl p-8"
      >
        {renderContent()}
        <div className="mt-6 text-center">
          <Link
            to="/admin/login"
            className="inline-flex items-center text-sm font-medium text-primary-600 hover:text-primary-500 dark:text-primary-400 dark:hover:text-primary-300"
          >
            <SafeIcon icon={FiArrowLeft} className="mr-2 h-4 w-4" />
            Back to Sign In
          </Link>
        </div>
      </motion.div>
    </div>
  );
};

export default ResetPassword;
