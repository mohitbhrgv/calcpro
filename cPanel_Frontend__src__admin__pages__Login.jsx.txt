import React, { useState, useEffect } from "react";
import { Navigate, Link } from "react-router-dom";
import { motion, AnimatePresence } from "framer-motion";
import { useAuth } from "../context/AuthContext";
import { useTheme } from "../context/ThemeContext";
import { useLayout } from "../context/LayoutContext";
import SafeIcon from "../../common/SafeIcon";
import toast from "react-hot-toast";
import * as IoIcons from "react-icons/io5";
import * as HiIcons from "react-icons/hi";
import * as BiIcons from "react-icons/bi";

const Login = () => {
  const { user, login, setAuthenticatedSession } = useAuth();
  const { isDark, toggleTheme } = useTheme();
  const { setBackgroundClass } = useLayout();

  const [uiStep, setUiStep] = useState("credentials");
  const [formData, setFormData] = useState({
    email: "",
    password: "",
  });
  const [otpCode, setOtpCode] = useState("");
  const [showPassword, setShowPassword] = useState(false);
  const [loading, setLoading] = useState(false);
  const [isLocked, setIsLocked] = useState(false);
  const [lockoutMessage, setLockoutMessage] = useState("");
  const [initialCheckDone, setInitialCheckDone] = useState(false);
  const [countdown, setCountdown] = useState(60);
  const [canResend, setCanResend] = useState(false);

  useEffect(() => {
    // Professional Gradient Background
    setBackgroundClass(
      "bg-gradient-to-br from-slate-900 via-indigo-950 to-slate-900",
    );
    return () => {
      setBackgroundClass("bg-gray-50 dark:bg-gray-900");
    };
  }, [setBackgroundClass]);

  useEffect(() => {
    let timer;
    if (uiStep === "otp" && !canResend) {
      timer = setInterval(() => {
        setCountdown((prev) => {
          if (prev <= 1) {
            clearInterval(timer);
            setCanResend(true);
            return 0;
          }
          return prev - 1;
        });
      }, 1000);
    }
    return () => clearInterval(timer);
  }, [uiStep, canResend]);

  useEffect(() => {
    const verifyLockoutStatus = async () => {
      try {
        const response = await fetch(
          "http://localhost/calcpro-api/api/public/check_ip_status.php",
        );
        const result = await response.json();
        if (result.success && result.data.locked) {
          setIsLocked(true);
          setLockoutMessage(result.data.message);
          setUiStep("credentials");
        }
      } catch (error) {
        console.error("Could not verify IP status with the server.");
      } finally {
        setInitialCheckDone(true);
      }
    };
    verifyLockoutStatus();
  }, []);

  // --- Redirect Guard: Keep loading state visible until delay finishes ---
  if (user && !loading) {
    return <Navigate to="/admin/dashboard" replace />;
  }

  const handleSubmit = async (e) => {
    e.preventDefault();
    setLoading(true);
    const toastId = "login-toast";

    const minDelayPromise = new Promise((resolve) => setTimeout(resolve, 2000));

    try {
      // 1. Run Login & Timer concurrently.
      // Catch login errors here to prevent Promise.all from rejecting early.
      const [res] = await Promise.all([
        login(formData.email, formData.password).catch((err) => ({
          error: err,
        })),
        minDelayPromise,
      ]);

      // 2. Check for captured error
      if (res && res.error) {
        throw res.error;
      }

      // 3. Handle Success
      if (res && res.success) {
        if (res.data?.otp_required) {
          toast.success(res.message, { id: toastId });
          setUiStep("otp");
          setCanResend(false);
          setCountdown(60);
          setLoading(false); // Stop loading to render OTP form
        } else {
          toast.success(res.message, { id: toastId });
          setLoading(false); // Stop loading to trigger Redirect Guard
        }
      } else {
        // API returned success: false but didn't throw
        throw new Error(res?.message || "Login failed");
      }
    } catch (error) {
      // 4. Handle Errors (After 2s delay)
      if (error && error.message) {
        toast.error(error.message, { id: toastId });

        if (
          error.errorCode === "IP_TEMPBANNED" ||
          error.errorCode === "IP_PERMABANNED"
        ) {
          setIsLocked(true);
          setLockoutMessage(error.message);
          setUiStep("credentials");
        }
      } else {
        toast.error(
          "Could not connect to the server. Please check your network.",
          { id: toastId },
        );
      }
      setLoading(false); // Ensure loading stops
    }
  };

  const handleVerifySubmit = async (e) => {
    e.preventDefault();
    if (otpCode.length !== 6) {
      toast.error("Please enter a valid 6-digit OTP.", { id: "login-toast" });
      return;
    }
    setLoading(true);
    const toastId = "login-toast";

    const minDelayPromise = new Promise((resolve) => setTimeout(resolve, 2000));

    try {
      const apiCall = fetch(
        "http://localhost/calcpro-api/api/admin/verify_otp.php",
        {
          method: "POST",
          headers: { "Content-Type": "application/json" },
          body: JSON.stringify({ email: formData.email, otp_code: otpCode }),
        },
      ).then((res) => res.json());

      const [res] = await Promise.all([
        apiCall.catch((err) => ({ error: err })),
        minDelayPromise,
      ]);

      if (res && res.error) throw res.error;

      if (res.success) {
        setAuthenticatedSession(res.data);
        toast.success(res.message, { id: toastId });
        setLoading(false); // Stop loading to trigger Redirect Guard
      } else {
        toast.error(res.message, { id: toastId });

        if (
          res.errorCode === "IP_TEMPBANNED" ||
          res.errorCode === "IP_PERMABANNED"
        ) {
          setUiStep("credentials");
          setIsLocked(true);
          setLockoutMessage(res.message);
          setOtpCode("");
        }
        setLoading(false);
      }
    } catch (error) {
      toast.error("Could not connect to the server.", { id: toastId });
      setLoading(false);
    }
  };

  const handleResendCode = async () => {
    if (!canResend) return;
    setLoading(true);
    try {
      const response = await fetch(
        "http://localhost/calcpro-api/api/admin/resend_otp.php",
        {
          method: "POST",
          headers: { "Content-Type": "application/json" },
          body: JSON.stringify({ email: formData.email }),
        },
      );
      const result = await response.json();
      if (result.success) {
        toast.success(result.message, { id: "login-toast" });
        setCanResend(false);
        setCountdown(60);
      } else {
        toast.error(result.message, { id: "login-toast" });
      }
    } catch (error) {
      toast.error("Failed to resend code.", { id: "login-toast" });
    } finally {
      setLoading(false);
    }
  };

  const handleChange = (e) => {
    setFormData((prev) => ({ ...prev, [e.target.name]: e.target.value }));
  };

  // --- MODERN ALERT COMPONENT ---
  const renderLockoutContent = () => {
    if (!isLocked) return null;
    return (
      <motion.div
        initial={{ opacity: 0, scale: 0.95 }}
        animate={{ opacity: 1, scale: 1 }}
        transition={{ duration: 0.3 }}
        className="mb-6 bg-red-50 dark:bg-red-900/10 border border-red-200 dark:border-red-800 rounded-xl p-4 text-center shadow-inner"
      >
        <div className="flex flex-col items-center">
          <div className="p-2 bg-red-100 dark:bg-red-900/40 rounded-full mb-2">
            <SafeIcon
              icon={IoIcons.IoAlert}
              className="w-5 h-5 text-red-600 dark:text-red-400"
            />
          </div>
          <h3 className="text-sm font-bold text-red-800 dark:text-red-200 mb-1">
            Access Denied
          </h3>
          <p className="text-xs text-red-600 dark:text-red-300 leading-relaxed max-w-xs mx-auto">
            {lockoutMessage}
          </p>
        </div>
      </motion.div>
    );
  };

  if (!initialCheckDone) {
    return null;
  }

  return (
    <div className="flex flex-col items-center justify-center min-h-screen p-4 relative font-inter">
      {/* Dynamic Background Elements */}
      <div className="absolute top-0 left-0 w-full h-full overflow-hidden z-0">
        <div className="absolute top-[20%] left-[20%] w-72 h-72 bg-indigo-500/20 rounded-full blur-[100px]"></div>
        <div className="absolute bottom-[20%] right-[20%] w-80 h-80 bg-blue-500/10 rounded-full blur-[100px]"></div>
      </div>

      <div className="absolute top-6 right-6 z-20">
        <motion.button
          whileHover={{ scale: 1.1 }}
          whileTap={{ scale: 0.9 }}
          onClick={toggleTheme}
          className="p-3 bg-white/10 dark:bg-black/20 backdrop-blur-md rounded-full hover:bg-white/20 dark:hover:bg-black/30 transition-colors border border-white/10"
        >
          <SafeIcon
            icon={isDark ? IoIcons.IoSunnyOutline : IoIcons.IoMoonOutline}
            className="w-5 h-5 text-white"
          />
        </motion.button>
      </div>

      <motion.div
        initial={{ opacity: 0, y: 30 }}
        animate={{ opacity: 1, y: 0 }}
        transition={{ duration: 0.5, ease: "easeOut" }}
        className="w-full max-w-md z-10"
      >
        <div className="bg-white dark:bg-gray-800 rounded-3xl shadow-2xl p-8 md:p-10 border border-gray-100 dark:border-gray-700">
          <AnimatePresence mode="wait">
            {uiStep === "credentials" ? (
              <motion.div
                key="credentials"
                exit={{ opacity: 0, x: -20 }}
                transition={{ duration: 0.2 }}
              >
                <div className="text-center mb-8">
                  <div className="w-16 h-16 bg-gradient-to-br from-indigo-600 to-blue-600 rounded-2xl mx-auto mb-5 flex items-center justify-center shadow-lg shadow-indigo-500/30">
                    <span className="text-white font-bold text-2xl">C</span>
                  </div>
                  <h1 className="text-2xl font-bold text-gray-900 dark:text-white font-montserrat tracking-tight">
                    Welcome Back
                  </h1>
                  <p className="text-sm text-gray-500 dark:text-gray-400 mt-2">
                    Enter your credentials to access the admin panel.
                  </p>
                </div>

                {renderLockoutContent()}

                <form onSubmit={handleSubmit} className="space-y-5">
                  <fieldset
                    disabled={isLocked || loading}
                    className="space-y-5"
                  >
                    <div className="space-y-1.5">
                      <label className="block text-xs font-bold text-gray-700 dark:text-gray-300 uppercase tracking-wide ml-1">
                        Email Address
                      </label>
                      <div className="relative group">
                        <SafeIcon
                          icon={IoIcons.IoMailOutline}
                          className="absolute left-3.5 top-1/2 transform -translate-y-1/2 text-gray-400 group-focus-within:text-indigo-500 transition-colors w-5 h-5 z-10"
                        />
                        <input
                          type="email"
                          name="email"
                          value={formData.email}
                          onChange={handleChange}
                          required
                          className="w-full pl-11 pr-4 py-3 border border-gray-300 dark:border-gray-600 rounded-xl focus:ring-2 focus:ring-indigo-500/20 focus:border-indigo-500 bg-white dark:bg-gray-900 text-gray-900 dark:text-white placeholder-gray-400 transition-all outline-none text-sm shadow-sm"
                          placeholder="Email Address"
                        />
                      </div>
                    </div>
                    <div className="space-y-1.5">
                      <div className="flex justify-between items-center ml-1">
                        <label className="block text-xs font-bold text-gray-700 dark:text-gray-300 uppercase tracking-wide">
                          Password
                        </label>
                        <Link
                          to="/admin/reset-password"
                          className="text-xs font-medium text-indigo-600 dark:text-indigo-400 hover:text-indigo-700 dark:hover:text-indigo-300 transition-colors"
                        >
                          Forgot password?
                        </Link>
                      </div>
                      <div className="relative group">
                        <SafeIcon
                          icon={IoIcons.IoLockClosedOutline}
                          className="absolute left-3.5 top-1/2 transform -translate-y-1/2 text-gray-400 group-focus-within:text-indigo-500 transition-colors w-5 h-5 z-10"
                        />
                        <input
                          type={showPassword ? "text" : "password"}
                          name="password"
                          value={formData.password}
                          onChange={handleChange}
                          required
                          className="w-full pl-11 pr-12 py-3 border border-gray-300 dark:border-gray-600 rounded-xl focus:ring-2 focus:ring-indigo-500/20 focus:border-indigo-500 bg-white dark:bg-gray-900 text-gray-900 dark:text-white placeholder-gray-400 transition-all outline-none text-sm shadow-sm"
                          placeholder="Password"
                        />
                        <button
                          type="button"
                          onClick={() => setShowPassword(!showPassword)}
                          className="absolute right-3.5 top-1/2 transform -translate-y-1/2 text-gray-400 hover:text-gray-600 dark:hover:text-gray-300 focus:outline-none p-1 rounded-md transition-colors"
                        >
                          <SafeIcon
                            icon={
                              showPassword ? HiIcons.HiEyeOff : HiIcons.HiEye
                            }
                            className="w-5 h-5"
                          />
                        </button>
                      </div>
                    </div>
                  </fieldset>

                  <motion.button
                    whileHover={{ scale: 1.01 }}
                    whileTap={{ scale: 0.99 }}
                    type="submit"
                    disabled={loading || isLocked}
                    className="w-full bg-indigo-600 hover:bg-indigo-700 text-white font-bold py-3 px-4 rounded-xl transition-all shadow-lg shadow-indigo-500/30 disabled:opacity-50 disabled:cursor-not-allowed disabled:shadow-none flex items-center justify-center space-x-2 text-sm mt-2"
                  >
                    {loading ? (
                      <>
                        <span className="animate-spin rounded-full h-4 w-4 border-2 border-white border-t-transparent"></span>
                        <span>Verifying Credentials...</span>
                      </>
                    ) : (
                      <>
                        <SafeIcon icon={BiIcons.BiLogIn} className="w-5 h-5" />
                        <span>Sign In</span>
                      </>
                    )}
                  </motion.button>
                </form>
              </motion.div>
            ) : (
              <motion.div
                key="otp"
                initial={{ opacity: 0, x: 20 }}
                animate={{ opacity: 1, x: 0 }}
                transition={{ duration: 0.2 }}
              >
                <div className="text-center mb-8">
                  <div className="w-16 h-16 bg-indigo-50 dark:bg-indigo-900/30 rounded-full mx-auto mb-4 flex items-center justify-center">
                    <SafeIcon
                      icon={IoIcons.IoShieldCheckmarkOutline}
                      className="w-8 h-8 text-indigo-600 dark:text-indigo-400"
                    />
                  </div>
                  <h1 className="text-2xl font-bold text-gray-900 dark:text-white mb-2">
                    Security Verification
                  </h1>
                  <p className="text-sm text-gray-500 dark:text-gray-400 max-w-[260px] mx-auto">
                    We've sent a 6-digit code to <br />
                    <span className="font-semibold text-indigo-600 dark:text-indigo-400">
                      {formData.email}
                    </span>
                  </p>
                </div>
                <form onSubmit={handleVerifySubmit} className="space-y-6">
                  <div>
                    <input
                      type="text"
                      value={otpCode}
                      onChange={(e) =>
                        setOtpCode(
                          e.target.value.replace(/\D/g, "").slice(0, 6),
                        )
                      }
                      maxLength="6"
                      className="w-full text-center tracking-[0.5em] text-2xl font-bold py-4 border border-gray-300 dark:border-gray-600 rounded-xl focus:ring-2 focus:ring-indigo-500/20 focus:border-indigo-500 bg-white dark:bg-gray-900 text-gray-900 dark:text-white outline-none transition-all shadow-sm"
                      placeholder="Enter 6-digit code"
                      autoFocus
                    />
                  </div>
                  <motion.button
                    whileHover={{ scale: 1.01 }}
                    whileTap={{ scale: 0.99 }}
                    type="submit"
                    disabled={loading || otpCode.length < 6}
                    className="w-full bg-indigo-600 hover:bg-indigo-700 text-white font-bold py-3 px-4 rounded-xl transition-all shadow-lg shadow-indigo-500/30 disabled:opacity-50 disabled:cursor-not-allowed disabled:shadow-none"
                  >
                    {loading ? (
                      <span className="flex items-center justify-center gap-2">
                        <span className="animate-spin rounded-full h-4 w-4 border-2 border-white border-t-transparent"></span>
                        Verifying...
                      </span>
                    ) : (
                      <span>Verify Access</span>
                    )}
                  </motion.button>

                  <div className="text-center space-y-4">
                    <div className="text-sm">
                      {canResend ? (
                        <button
                          type="button"
                          onClick={handleResendCode}
                          disabled={loading}
                          className="font-medium text-indigo-600 hover:text-indigo-700 dark:text-indigo-400 dark:hover:text-indigo-300 hover:underline transition-colors"
                        >
                          Resend Verification Code
                        </button>
                      ) : (
                        <span className="text-gray-400 dark:text-gray-500">
                          Resend code in {countdown}s
                        </span>
                      )}
                    </div>

                    <div className="pt-4 border-t border-gray-100 dark:border-gray-700">
                      <button
                        type="button"
                        onClick={() => setUiStep("credentials")}
                        className="text-xs font-medium text-gray-500 hover:text-gray-700 dark:text-gray-400 dark:hover:text-gray-200 transition-colors flex items-center justify-center gap-1 mx-auto group"
                      >
                        <SafeIcon
                          icon={HiIcons.HiArrowLeft}
                          className="w-3 h-3 transition-transform group-hover:-translate-x-1"
                        />
                        Back to login
                      </button>
                    </div>
                  </div>
                </form>
              </motion.div>
            )}
          </AnimatePresence>
        </div>
      </motion.div>
    </div>
  );
};

export default Login;
