<?php
// Core utilities
require_once '../../core/config.php';
require_once '../../core/response.php';
require_once '../../core/auth.php'; 

// --- Security Check ---
// Ensures only a logged-in admin can perform this action.
$admin_data = verifyAdminToken();

try {
    // --- 1. Get Incoming Data ---
    $json = file_get_contents('php://input');
    $data = json_decode($json);

    // --- 2. Validation ---
    // We need the ID of the calculator to toggle.
    if (!isset($data->id) || !is_numeric($data->id)) {
        sendResponse(false, null, 'A valid Calculator ID is required.', 400, 'INVALID_INPUT');
    }

    $id = (int)$data->id;

    // --- 3. The SQL UPDATE Query ---
    // This query is very simple and efficient. It finds the calculator by its ID
    // and sets the `is_featured` column to the opposite of its current value.
    // If it's 1 (true), `NOT 1` becomes 0 (false).
    // If it's 0 (false), `NOT 0` becomes 1 (true).
    $sql = "UPDATE calculators SET is_featured = NOT is_featured WHERE id = ?";
    $stmt = $conn->prepare($sql);
    
    if ($stmt === false) {
        throw new Exception('Failed to prepare the SQL statement.');
    }

    $stmt->bind_param("i", $id);

    // --- 4. Execute and Check for Success ---
    if ($stmt->execute()) {
        if ($stmt->affected_rows > 0) {
            // --- 5. Fetch the New Status ---
            // To provide a good user experience, we send back the new status (true or false).
            // The frontend UI can use this to confirm the state of the toggle without needing to refresh everything.
            $select_sql = "SELECT is_featured FROM calculators WHERE id = ?";
            $select_stmt = $conn->prepare($select_sql);
            $select_stmt->bind_param("i", $id);
            $select_stmt->execute();
            $result = $select_stmt->get_result()->fetch_assoc();
            $select_stmt->close();
            
            $new_status = (bool)$result['is_featured'];
            
            sendResponse(true, ['is_featured' => $new_status], 'Featured status updated successfully.', 200);
        } else {
            sendResponse(false, null, 'Calculator not found.', 404, 'NOT_FOUND');
        }
    } else {
        throw new Exception("Database update failed: " . $stmt->error);
    }
    
    $stmt->close();

} catch (Exception $e) {
    error_log("Error in toggle_featured_calculator.php: " . $e->getMessage());
    sendResponse(false, null, 'An internal server error occurred.', 500, 'SERVER_ERROR');
}

$conn->close();
?>
