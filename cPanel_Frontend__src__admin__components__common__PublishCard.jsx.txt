import React, { useState } from "react";
import SafeIcon from "../../../common/SafeIcon";
import * as FiIcons from "react-icons/fi";
import { motion, AnimatePresence } from "framer-motion";
import { useAdminCache } from "../../context/AdminCacheContext";
import { formatDateTime } from "../../utils/dateUtils";

const { FiChevronDown, FiChevronUp, FiCalendar, FiClock, FiLink } = FiIcons;

const PublishCard = ({
  status,
  slug,
  onSlugChange,
  onStatusChange,
  createdAt,
  updatedAt,
}) => {
  const [isExpanded, setIsExpanded] = useState(true);

  // Access global settings for timezone/date formats
  const { getCache } = useAdminCache();
  const settings = getCache("settings_general") || {};

  const showUpdatedField = () => {
    if (!createdAt || !updatedAt) return false;
    const created = new Date(createdAt).getTime();
    const updated = new Date(updatedAt).getTime();
    // Show if updated more than 1 minute after creation
    return updated - created > 60000;
  };

  return (
    <div className="bg-white dark:bg-gray-800 rounded-xl shadow-sm border border-gray-200 dark:border-gray-700 overflow-hidden">
      <button
        type="button"
        onClick={() => setIsExpanded(!isExpanded)}
        className="w-full flex items-center justify-between p-4 bg-gray-50 dark:bg-gray-700/50 border-b border-gray-200 dark:border-gray-700 transition-colors hover:bg-gray-100 dark:hover:bg-gray-700"
      >
        <h3 className="font-semibold text-sm uppercase tracking-wide text-gray-700 dark:text-gray-300">
          Publishing & Url
        </h3>
        <SafeIcon
          icon={isExpanded ? FiChevronUp : FiChevronDown}
          className="w-4 h-4 text-gray-500"
        />
      </button>

      <AnimatePresence>
        {isExpanded && (
          <motion.div
            initial={{ height: 0, opacity: 0 }}
            animate={{ height: "auto", opacity: 1 }}
            exit={{ height: 0, opacity: 0 }}
            className="p-4 space-y-5"
          >
            {/* 1. Status Dropdown */}
            <div>
              <label className="block text-xs font-bold text-gray-500 uppercase mb-1">
                Status
              </label>
              <div className="relative">
                <select
                  name="status"
                  value={status}
                  onChange={(e) => onStatusChange(e.target.value)}
                  className={`w-full p-2 border rounded-lg text-sm focus:ring-2 outline-none transition-all bg-white dark:bg-gray-700 dark:text-white appearance-none ${
                    status === "published"
                      ? "border-green-500 focus:ring-green-200 dark:focus:ring-green-900 text-green-700 dark:text-green-400 font-medium"
                      : "border-gray-300 dark:border-gray-600 focus:ring-blue-200 dark:focus:ring-blue-900 text-gray-700"
                  }`}
                >
                  <option value="draft">Draft</option>
                  <option value="published">Published</option>
                </select>
                <div
                  className={`absolute right-3 top-1/2 -translate-y-1/2 w-2 h-2 rounded-full ${
                    status === "published" ? "bg-green-500" : "bg-amber-500"
                  }`}
                ></div>
              </div>
            </div>

            {/* 2. Permalink Input */}
            <div>
              <label className="block text-xs font-bold text-gray-500 uppercase mb-1">
                URL Slug
              </label>
              <div className="flex items-center border border-gray-300 dark:border-gray-600 rounded-lg overflow-hidden bg-white dark:bg-gray-700 focus-within:ring-2 focus-within:ring-blue-500 transition-shadow">
                <div className="pl-3 pr-2 text-gray-400 bg-gray-50 dark:bg-gray-800 border-r border-gray-200 dark:border-gray-600 h-9 flex items-center">
                  <SafeIcon icon={FiLink} className="w-3.5 h-3.5" />
                </div>
                <input
                  type="text"
                  value={slug || ""}
                  onChange={onSlugChange}
                  placeholder="page-title"
                  className="w-full p-2 text-sm border-none focus:ring-0 bg-transparent dark:text-white font-mono"
                />
              </div>
            </div>

            {/* 3. Metadata */}
            <div className="space-y-2 pt-2 border-t border-gray-100 dark:border-gray-700">
              <div className="flex items-center space-x-2 text-xs text-gray-500 dark:text-gray-400">
                <SafeIcon icon={FiCalendar} className="w-3 h-3 flex-shrink-0" />
                <span>Created:</span>
                <span className="font-mono text-gray-700 dark:text-gray-300 ml-auto">
                  {createdAt ? formatDateTime(createdAt, settings) : "Unsaved"}
                </span>
              </div>
              {showUpdatedField() && (
                <div className="flex items-center space-x-2 text-xs text-gray-500 dark:text-gray-400">
                  <SafeIcon icon={FiClock} className="w-3 h-3 flex-shrink-0" />
                  <span>Modified:</span>
                  <span className="font-mono text-gray-700 dark:text-gray-300 ml-auto">
                    {formatDateTime(updatedAt, settings)}
                  </span>
                </div>
              )}
            </div>
          </motion.div>
        )}
      </AnimatePresence>
    </div>
  );
};

export default PublishCard;
