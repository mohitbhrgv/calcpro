import React, { useState, useEffect } from "react";
import { useParams, useNavigate, Link } from "react-router-dom";
import { motion } from "framer-motion";
import toast from "react-hot-toast";
import SafeIcon from "../../common/SafeIcon";
import * as FiIcons from "react-icons/fi";

const { FiCheckCircle, FiAlertCircle, FiLogIn } = FiIcons;

const UnlockAccount = () => {
  const { token } = useParams();
  const navigate = useNavigate();
  const [status, setStatus] = useState("verifying"); // 'verifying', 'success', 'error'
  const [message, setMessage] = useState("Verifying your unlock request...");

  useEffect(() => {
    if (!token) {
      setStatus("error");
      setMessage(
        "No unlock token provided. Please use the link from your email."
      );
      return;
    }

    const verifyToken = async () => {
      try {
        const response = await fetch(
          "http://localhost/calcpro-api/api/public/unlock_account.php",
          {
            method: "POST",
            headers: { "Content-Type": "application/json" },
            body: JSON.stringify({ token }),
          }
        );

        const result = await response.json();

        if (result.success) {
          setStatus("success");
          setMessage(result.message);
          toast.success(result.message);
        } else {
          setStatus("error");
          setMessage(result.message || "An unknown error occurred.");
          toast.error(result.message || "An unknown error occurred.");
        }
      } catch (error) {
        setStatus("error");
        setMessage(
          "Could not connect to the server. Please check your connection and try again."
        );
        toast.error("Could not connect to the server.");
      }
    };

    // Delay verification slightly to allow UI to render first
    const timer = setTimeout(verifyToken, 1000);
    return () => clearTimeout(timer);
  }, [token]);

  const StatusIcon = () => {
    switch (status) {
      case "success":
        return (
          <SafeIcon icon={FiCheckCircle} className="w-12 h-12 text-green-500" />
        );
      case "error":
        return (
          <SafeIcon icon={FiAlertCircle} className="w-12 h-12 text-red-500" />
        );
      default: // 'verifying'
        return (
          <div className="w-12 h-12 border-4 border-blue-500 border-t-transparent rounded-full animate-spin"></div>
        );
    }
  };

  return (
    <div className="min-h-screen bg-gradient-to-br from-indigo-600 to-blue-500 flex items-center justify-center px-4">
      <motion.div
        initial={{ opacity: 0, y: 20 }}
        animate={{ opacity: 1, y: 0 }}
        transition={{ duration: 0.5 }}
        className="w-full max-w-md bg-white dark:bg-gray-800 rounded-xl shadow-2xl p-8 text-center"
      >
        <div className="w-20 h-20 bg-gradient-to-br from-indigo-100 to-blue-100 dark:from-indigo-900/20 dark:to-blue-900/20 rounded-full mx-auto mb-6 flex items-center justify-center">
          <StatusIcon />
        </div>

        <h1 className="text-2xl font-bold text-gray-900 dark:text-white mb-4">
          {status === "verifying" && "Unlocking Account..."}
          {status === "success" && "Account Unlocked!"}
          {status === "error" && "Unlock Failed"}
        </h1>

        <p className="text-gray-600 dark:text-gray-400 mb-8">{message}</p>

        {status !== "verifying" && (
          <motion.div
            initial={{ opacity: 0 }}
            animate={{ opacity: 1 }}
            transition={{ delay: 0.3 }}
          >
            <Link
              to="/admin/login"
              className="w-full bg-gradient-to-r from-indigo-600 to-blue-500 hover:from-indigo-700 hover:to-blue-600 text-white font-medium py-3 px-4 rounded-lg transition-colors flex items-center justify-center space-x-2"
            >
              <SafeIcon icon={FiLogIn} className="w-5 h-5" />
              <span>Proceed to Login</span>
            </Link>
          </motion.div>
        )}
      </motion.div>
    </div>
  );
};

export default UnlockAccount;
