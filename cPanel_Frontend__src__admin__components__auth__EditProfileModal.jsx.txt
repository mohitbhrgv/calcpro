
import React, { useState, useRef, useEffect } from "react";
import { useAuth } from "../../context/AuthContext";
import Modal from "../common/Modal";
import SafeIcon from "../../../common/SafeIcon";
import * as FiIcons from "react-icons/fi";
import toast from "react-hot-toast";
import { authenticatedFetch } from "../../utils/apiService";

const { FiUser, FiMail, FiLock, FiCamera, FiSave } = FiIcons;

const EditProfileModal = ({ isOpen, onClose }) => {
  const { user, updateUser } = useAuth();
  const [activeTab, setActiveTab] = useState("profile");
  const [isSubmitting, setIsSubmitting] = useState(false);
  const fileInputRef = useRef(null);

  const [formData, setFormData] = useState({
    name: "",
    email: "",
    currentPassword: "",
    newPassword: "",
    confirmPassword: "",
  });

  useEffect(() => {
    if (user) {
      setFormData({
        name: user.name || "",
        email: user.email || "",
        currentPassword: "",
        newPassword: "",
        confirmPassword: "",
      });
      setActiveTab("profile");
    }
  }, [isOpen, user]);

  const handleChange = (e) => {
    const { name, value } = e.target;
    setFormData((prev) => ({ ...prev, [name]: value }));
  };

  const handleProfileUpdate = async (e) => {
    e.preventDefault();
    setIsSubmitting(true);
    const toastId = "profile-update-toast";
    toast.loading("Updating profile...", { id: toastId });

    try {
      const result = await authenticatedFetch(
        "http://localhost/calcpro-api/api/admin/update_admin_profile.php",
        {
          method: "POST",
          body: JSON.stringify({
            name: formData.name,
            email: formData.email,
          }),
        }
      );

      if (result.success) {
        updateUser(result.data);
        toast.success(result.message, { id: toastId });
        onClose();
      } else {
        toast.error(result.message || "An unknown error occurred.", {
          id: toastId,
        });
      }
    } catch (error) {
      if (error.message !== "Unauthorized") {
        toast.error("Could not connect to the server.", { id: toastId });
      }
    } finally {
      setIsSubmitting(false);
    }
  };

  const handlePasswordUpdate = async (e) => {
    e.preventDefault();
    if (formData.newPassword !== formData.confirmPassword) {
      toast.error("New passwords do not match.");
      return;
    }
    if (formData.newPassword.length < 8) {
      toast.error("New password must be at least 8 characters long.");
      return;
    }

    setIsSubmitting(true);
    const toastId = "password-update-toast";
    toast.loading("Updating password...", { id: toastId });

    try {
      const result = await authenticatedFetch(
        "http://localhost/calcpro-api/api/admin/update_admin_password.php",
        {
          method: "POST",
          body: JSON.stringify({
            currentPassword: formData.currentPassword,
            newPassword: formData.newPassword,
          }),
        }
      );

      if (result.success) {
        toast.success(result.message, { id: toastId });
        onClose();
      } else {
        toast.error(result.message || "An unknown error occurred.", {
          id: toastId,
        });
      }
    } catch (error) {
      if (error.message !== "Unauthorized") {
        toast.error("Could not connect to the server.", { id: toastId });
      }
    } finally {
      setIsSubmitting(false);
    }
  };

  const handleProfilePictureChange = (e) => {
    const file = e.target.files[0];
    if (file) {
      console.log("Selected file:", file.name);
      toast.success(
        "Profile picture upload is not implemented in this version."
      );
    }
  };

  return (
    <Modal 
      isOpen={isOpen} 
      onClose={onClose} 
      title="Edit Profile" 
      size="lg"
      disableAnimation={true} // --- ANIMATION DISABLED ---
    >
      <div className="flex flex-col md:flex-row gap-8">
        {/* Left Side: Avatar and User Info */}
        <div className="md:w-1/3 text-center flex flex-col items-center">
          <div className="relative w-32 h-32 mx-auto group mb-4">
            <img
              src={
                user?.avatar ||
                `https://ui-avatars.com/api/?name=${user?.name.replace(
                  /\s/g,
                  "+"
                )}&background=1E3A8A&color=fff&size=128`
              }
              alt={user?.name}
              className="w-full h-full rounded-full object-cover border-4 border-white dark:border-gray-700 shadow-md"
            />
            <button
              onClick={() => fileInputRef.current.click()}
              className="absolute inset-0 bg-black/50 rounded-full flex items-center justify-center text-white opacity-0 group-hover:opacity-100 transition-opacity"
              aria-label="Change profile picture"
            >
              <SafeIcon icon={FiCamera} className="w-6 h-6" />
            </button>
            <input
              type="file"
              ref={fileInputRef}
              className="hidden"
              accept="image/png, image/jpeg"
              onChange={handleProfilePictureChange}
            />
          </div>
          <h3 className="text-xl font-bold text-gray-900 dark:text-white">
            {user?.name}
          </h3>
          <p className="text-sm text-gray-500 dark:text-gray-400">
            {user?.email}
          </p>
        </div>

        {/* Right Side: Tabbed Form */}
        <div className="md:w-2/3">
          <div className="border-b border-gray-200 dark:border-gray-700">
            <nav className="flex space-x-4">
              <button
                onClick={() => setActiveTab("profile")}
                className={`px-3 py-2 text-sm font-medium transition-colors ${
                  activeTab === "profile"
                    ? "border-b-2 border-calcpro-primary text-calcpro-primary dark:text-blue-400 dark:border-blue-400"
                    : "border-transparent text-gray-500 dark:text-gray-400 hover:text-gray-700 dark:hover:text-gray-200"
                }`}
              >
                Profile Details
              </button>
              <button
                onClick={() => setActiveTab("password")}
                className={`px-3 py-2 text-sm font-medium transition-colors ${
                  activeTab === "password"
                    ? "border-b-2 border-calcpro-primary text-calcpro-primary dark:text-blue-400 dark:border-blue-400"
                    : "border-transparent text-gray-500 dark:text-gray-400 hover:text-gray-700 dark:hover:text-gray-200"
                }`}
              >
                Change Password
              </button>
            </nav>
          </div>

          <div className="py-6">
            {activeTab === "profile" && (
              <form onSubmit={handleProfileUpdate} className="space-y-6">
                <div>
                  <label className="block text-sm font-medium mb-2 text-gray-700 dark:text-gray-300">
                    Full Name
                  </label>
                  <div className="relative">
                    <SafeIcon
                      icon={FiUser}
                      className="absolute left-3 top-1/2 -translate-y-1/2 h-5 w-5 text-gray-400 dark:text-gray-500"
                    />
                    <input
                      type="text"
                      name="name"
                      value={formData.name}
                      onChange={handleChange}
                      required
                      className="w-full pl-10 pr-4 py-2 border border-gray-300 dark:border-gray-600 rounded-lg bg-white dark:bg-gray-700 text-gray-900 dark:text-white placeholder-gray-400 dark:placeholder-gray-500 focus:ring-2 focus:ring-calcpro-primary focus:border-transparent transition-colors"
                    />
                  </div>
                </div>
                <div>
                  <label className="block text-sm font-medium mb-2 text-gray-700 dark:text-gray-300">
                    Email Address
                  </label>
                  <div className="relative">
                    <SafeIcon
                      icon={FiMail}
                      className="absolute left-3 top-1/2 -translate-y-1/2 h-5 w-5 text-gray-400 dark:text-gray-500"
                    />
                    <input
                      type="email"
                      name="email"
                      value={formData.email}
                      onChange={handleChange}
                      required
                      className="w-full pl-10 pr-4 py-2 border border-gray-300 dark:border-gray-600 rounded-lg bg-white dark:bg-gray-700 text-gray-900 dark:text-white placeholder-gray-400 dark:placeholder-gray-500 focus:ring-2 focus:ring-calcpro-primary focus:border-transparent transition-colors"
                    />
                  </div>
                </div>
                <div className="flex justify-end pt-4">
                  <button
                    type="submit"
                    disabled={isSubmitting}
                    className="flex items-center space-x-2 bg-calcpro-primary text-white px-4 py-2 rounded-lg hover:bg-calcpro-primary/90 disabled:opacity-60 transition-colors"
                  >
                    <SafeIcon icon={FiSave} />
                    <span>{isSubmitting ? "Saving..." : "Save Changes"}</span>
                  </button>
                </div>
              </form>
            )}

            {activeTab === "password" && (
              <form onSubmit={handlePasswordUpdate} className="space-y-6">
                <div>
                  <label className="block text-sm font-medium mb-2 text-gray-700 dark:text-gray-300">
                    Current Password
                  </label>
                  <div className="relative">
                    <SafeIcon
                      icon={FiLock}
                      className="absolute left-3 top-1/2 -translate-y-1/2 h-5 w-5 text-gray-400 dark:text-gray-500"
                    />
                    <input
                      type="password"
                      name="currentPassword"
                      value={formData.currentPassword}
                      onChange={handleChange}
                      required
                      className="w-full pl-10 pr-4 py-2 border border-gray-300 dark:border-gray-600 rounded-lg bg-white dark:bg-gray-700 text-gray-900 dark:text-white placeholder-gray-400 dark:placeholder-gray-500 focus:ring-2 focus:ring-calcpro-primary focus:border-transparent transition-colors"
                    />
                  </div>
                </div>
                <div>
                  <label className="block text-sm font-medium mb-2 text-gray-700 dark:text-gray-300">
                    New Password
                  </label>
                  <div className="relative">
                    <SafeIcon
                      icon={FiLock}
                      className="absolute left-3 top-1/2 -translate-y-1/2 h-5 w-5 text-gray-400 dark:text-gray-500"
                    />
                    <input
                      type="password"
                      name="newPassword"
                      value={formData.newPassword}
                      onChange={handleChange}
                      required
                      className="w-full pl-10 pr-4 py-2 border border-gray-300 dark:border-gray-600 rounded-lg bg-white dark:bg-gray-700 text-gray-900 dark:text-white placeholder-gray-400 dark:placeholder-gray-500 focus:ring-2 focus:ring-calcpro-primary focus:border-transparent transition-colors"
                    />
                  </div>
                </div>
                <div>
                  <label className="block text-sm font-medium mb-2 text-gray-700 dark:text-gray-300">
                    Confirm New Password
                  </label>
                  <div className="relative">
                    <SafeIcon
                      icon={FiLock}
                      className="absolute left-3 top-1/2 -translate-y-1/2 h-5 w-5 text-gray-400 dark:text-gray-500"
                    />
                    <input
                      type="password"
                      name="confirmPassword"
                      value={formData.confirmPassword}
                      onChange={handleChange}
                      required
                      className="w-full pl-10 pr-4 py-2 border border-gray-300 dark:border-gray-600 rounded-lg bg-white dark:bg-gray-700 text-gray-900 dark:text-white placeholder-gray-400 dark:placeholder-gray-500 focus:ring-2 focus:ring-calcpro-primary focus:border-transparent transition-colors"
                    />
                  </div>
                </div>
                <div className="flex justify-end pt-4">
                  <button
                    type="submit"
                    disabled={isSubmitting}
                    className="flex items-center space-x-2 bg-calcpro-primary text-white px-4 py-2 rounded-lg hover:bg-calcpro-primary/90 disabled:opacity-60 transition-colors"
                  >
                    <SafeIcon icon={FiSave} />
                    <span>
                      {isSubmitting ? "Updating..." : "Update Password"}
                    </span>
                  </button>
                </div>
              </form>
            )}
          </div>
        </div>
      </div>
    </Modal>
  );
};

export default EditProfileModal;
