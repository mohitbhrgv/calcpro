import Link from 'next/link';
import { Metadata } from 'next';
import NotFoundClient from '@/components/common/NotFoundClient';
import SafeIcon from '@/components/common/SafeIcon';
import * as FiIcons from 'react-icons/fi';
import { getSiteSettings } from '@/lib/content';
import { constructMetadata } from '@/lib/seo';

const { FiArrowLeft, FiAlertTriangle } = FiIcons;

export async function generateMetadata(): Promise<Metadata> {
  const settings = await getSiteSettings();
  
  // Use the constructMetadata logic with type '404' to apply 
  // the "seo_special_404_title" template from cPanel
  return constructMetadata({
    data: { title: "Page Not Found" },
    settings,
    type: "404",
    path: "/404"
  });
}

export default function NotFound() {
  return (
    <div className="bg-neutral-50 dark:bg-neutral-900 py-20 min-h-[70vh] flex items-center justify-center font-inter">
      {/* Client component to handle API logging */}
      <NotFoundClient />
      
      <div className="text-center px-4">
        <div className="mb-6 flex justify-center">
            <div className="p-4 bg-red-50 dark:bg-red-900/20 rounded-full text-red-500 dark:text-red-400">
                <SafeIcon icon={FiAlertTriangle} className="w-12 h-12" />
            </div>
        </div>
        
        <h1 className="text-6xl font-extrabold text-neutral-900 dark:text-white mb-2 tracking-tight">
            404
        </h1>
        <h2 className="text-2xl font-bold text-neutral-800 dark:text-gray-200 mb-4">
            Page Not Found
        </h2>
        
        <p className="text-neutral-600 dark:text-neutral-400 mb-8 max-w-md mx-auto text-lg">
          The page you are looking for does not exist, has been moved, or is temporarily unavailable.
        </p>
        
        <Link
          href="/"
          className="inline-flex items-center px-8 py-3.5 bg-indigo-600 hover:bg-indigo-700 text-white rounded-xl transition-all shadow-lg shadow-indigo-500/30 font-bold hover:-translate-y-0.5"
        >
          <SafeIcon icon={FiArrowLeft} className="w-5 h-5 mr-2" />
          <span>Return Home</span>
        </Link>
      </div>
    </div>
  );
}
