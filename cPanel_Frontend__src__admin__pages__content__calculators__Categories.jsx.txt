
import React, { useState, useEffect, useCallback, useMemo } from "react";
import { motion, AnimatePresence } from "framer-motion";
import DataTable from "../../../components/common/DataTable";
import Modal from "../../../components/common/Modal";
import SafeIcon from "../../../../common/SafeIcon";
import * as FiIcons from "react-icons/fi";
import toast from "react-hot-toast";
import { authenticatedFetch } from "../../../utils/apiService";
import LoadingSpinner from "../../../components/common/LoadingSpinner";
import { useAdminCache } from "../../../context/AdminCacheContext";

const {
  FiPlus,
  FiGrid,
  FiAlertCircle,
  FiSearch,
  FiTrash2,
  FiAlertTriangle,
  FiGlobe,
  FiChevronDown,
  FiChevronUp,
  FiList,
  FiFolder,
} = FiIcons;

// Helper component for consistent scrollable cells
const ScrollableCell = ({
  text,
  colorClass = "text-gray-600 dark:text-gray-300",
}) => (
  <div
    className={`text-xs ${colorClass} max-h-[60px] overflow-y-auto whitespace-normal break-words scrollbar-thin scrollbar-thumb-gray-200 dark:scrollbar-thumb-gray-600 pr-1 leading-snug`}
  >
    {text}
  </div>
);

// Helper component for wrapped text cells
const WrappedCell = ({ text, isSlug = false }) => (
  <div
    className={`text-sm text-gray-900 dark:text-white whitespace-normal ${
      isSlug ? "break-all font-mono text-xs text-gray-500" : "break-words"
    } max-w-[180px] leading-snug`}
  >
    {text}
  </div>
);

const CalculatorCategories = () => {
  const { getCache, setCacheData, invalidateCache, isInitialLoadComplete } =
    useAdminCache();
  const cacheKey = "calculator_categories";

  const cachedData = getCache(cacheKey);
  const [allCategories, setAllCategories] = useState(cachedData || []);

  const [isUpdating, setIsUpdating] = useState(false);
  const [showModal, setShowModal] = useState(false);
  const [editingCategory, setEditingCategory] = useState(null);
  const [isSubmitting, setIsSubmitting] = useState(false);

  // --- Bulk Selection ---
  const [selectedIds, setSelectedIds] = useState([]);
  const [isBulkDeleting, setIsBulkDeleting] = useState(false);

  // --- Delete Confirmation State ---
  const [isDeleteModalOpen, setIsDeleteModalOpen] = useState(false);
  const [categoryToDelete, setCategoryToDelete] = useState(null);
  const [confirmInput, setConfirmInput] = useState("");

  // --- Bulk Delete Confirmation ---
  const [isBulkDeleteModalOpen, setIsBulkDeleteModalOpen] = useState(false);

  const [formData, setFormData] = useState({
    name: "",
    slug: "",
    description: "",
    parent_id: "",
    meta_title: "",
    meta_description: "",
  });
  const [searchTerm, setSearchTerm] = useState("");
  const [showSeoFields, setShowSeoFields] = useState(false);

  const fetchCategories = useCallback(
    async (forceRefresh = false) => {
      if (!forceRefresh) {
        const cached = getCache(cacheKey);
        if (cached) {
          if (JSON.stringify(cached) !== JSON.stringify(allCategories)) {
            setAllCategories(cached);
          }
          return;
        }
      }

      setIsUpdating(true);

      try {
        const result = await authenticatedFetch(
          "http://localhost/calcpro-api/api/admin/get_calculator_categories.php"
        );
        if (result.success) {
          setAllCategories(result.data);
          setCacheData(cacheKey, result.data);
        } else {
          toast.error(result.message || "Failed to fetch categories.");
        }
      } catch (error) {
        if (error.message !== "Unauthorized") {
          toast.error("Could not connect to the server.");
        }
      } finally {
        setIsUpdating(false);
      }
    },
    [getCache, setCacheData, cacheKey, allCategories]
  );

  useEffect(() => {
    fetchCategories();
  }, [fetchCategories]);

  const filteredCategories = useMemo(() => {
    if (!searchTerm) return allCategories;
    return allCategories.filter(
      (category) =>
        category.name.toLowerCase().includes(searchTerm.toLowerCase()) ||
        category.slug.toLowerCase().includes(searchTerm.toLowerCase())
    );
  }, [allCategories, searchTerm]);

  const parentCategoryIds = useMemo(
    () => new Set(allCategories.map((c) => c.parent_id).filter(Boolean)),
    [allCategories]
  );

  // --- Bulk Handlers ---
  const handleSelectRow = (id) => {
    setSelectedIds((prev) =>
      prev.includes(id) ? prev.filter((i) => i !== id) : [...prev, id]
    );
  };

  const handleSelectAll = (e) => {
    setSelectedIds(e.target.checked ? filteredCategories.map((c) => c.id) : []);
  };

  const handleBulkDelete = () => {
    if (selectedIds.length === 0) return;
    setIsBulkDeleteModalOpen(true);
  };

  const confirmBulkDelete = async () => {
    if (confirmInput !== "DELETE") {
      toast.error("Please type DELETE to confirm.");
      return;
    }

    setIsBulkDeleting(true);
    try {
      const result = await authenticatedFetch(
        "http://localhost/calcpro-api/api/admin/delete_calculator_categories_bulk.php",
        {
          method: "POST",
          body: JSON.stringify({ ids: selectedIds }),
        }
      );

      if (result.success) {
        toast.success(result.message);
        setSelectedIds([]);
        invalidateCache(cacheKey);
        await fetchCategories(true);
      } else {
        toast.error(result.message);
      }
    } catch (error) {
      toast.error("Bulk delete failed.");
    } finally {
      setIsBulkDeleting(false);
      setIsBulkDeleteModalOpen(false);
      setConfirmInput("");
    }
  };

  const columns = [
    {
      key: "name",
      label: "Name",
      render: (val) => <WrappedCell text={val} />,
    },
    {
      key: "slug",
      label: "Slug",
      render: (val) => <WrappedCell text={val} isSlug={true} />,
    },
    {
      key: "parent_name",
      label: "Parent",
      render: (val, item) => {
        if (val) {
          return (
            <span className="inline-flex items-center px-2 py-1 rounded-md text-xs font-medium bg-blue-50 text-blue-700 dark:bg-blue-900/30 dark:text-blue-200 border border-blue-100 dark:border-blue-800 whitespace-normal text-center leading-tight max-w-[120px]">
              {val}
            </span>
          );
        }
        if (parentCategoryIds.has(item.id)) {
          return (
            <span className="inline-flex items-center px-2 py-1 rounded-md text-xs font-bold bg-emerald-50 text-emerald-700 dark:bg-emerald-900/30 dark:text-emerald-200 border border-emerald-100 dark:border-emerald-800">
              Is Parent
            </span>
          );
        }
        return <span className="text-gray-400 text-xs italic">Not Set</span>;
      },
    },
    {
      key: "description",
      label: "Description",
      render: (val) =>
        val ? (
          <ScrollableCell text={val} />
        ) : (
          <span className="text-gray-400 text-xs italic">Not Set</span>
        ),
    },
    {
      key: "calculator_count",
      label: "Calculators",
      render: (val) => (
        <span className="font-mono text-sm text-gray-600 dark:text-gray-300">
          {val}
        </span>
      ),
    },
  ];

  const handleSubmit = async (e) => {
    e.preventDefault();
    setIsSubmitting(true);

    const isEditing = editingCategory !== null;
    const url = isEditing
      ? "http://localhost/calcpro-api/api/admin/update_calculator_category.php"
      : "http://localhost/calcpro-api/api/admin/add_calculator_category.php";

    const finalParentId = hasChildren
      ? null
      : formData.parent_id === ""
        ? null
        : parseInt(formData.parent_id);

    const payload = {
      ...formData,
      parent_id: finalParentId,
    };

    if (isEditing) {
      payload.id = editingCategory.id;
    }

    try {
      const result = await authenticatedFetch(url, {
        method: "POST",
        body: JSON.stringify(payload),
      });

      if (result.success) {
        toast.success(result.message);
        handleCloseModal();
        invalidateCache(cacheKey);
        await fetchCategories(true);
      } else {
        toast.error(result.message || "An unknown error occurred.");
      }
    } catch (error) {
      if (error.message !== "Unauthorized") {
        toast.error("Could not connect to the server.");
      }
    } finally {
      setIsSubmitting(false);
    }
  };

  const handleEdit = (category) => {
    setEditingCategory(category);
    setFormData({
      name: category.name,
      slug: category.slug,
      description: category.description || "",
      parent_id: category.parent_id || "",
      meta_title: category.meta_title || "",
      meta_description: category.meta_description || "",
    });
    setShowSeoFields(!!(category.meta_title || category.meta_description));
    setShowModal(true);
  };

  const handleDelete = (category) => {
    setCategoryToDelete(category);
    setIsDeleteModalOpen(true);
  };

  const confirmDelete = async () => {
    if (!categoryToDelete || confirmInput !== "DELETE") {
      toast.error("Please type DELETE to confirm.");
      return;
    }

    try {
      const result = await authenticatedFetch(
        "http://localhost/calcpro-api/api/admin/delete_calculator_category.php",
        {
          method: "POST",
          body: JSON.stringify({ id: categoryToDelete.id }),
        }
      );

      if (result.success) {
        toast.success(result.message);
        invalidateCache(cacheKey);
        await fetchCategories(true);
      } else {
        toast.error(result.message || "Failed to delete category.");
      }
    } catch (error) {
      if (error.message !== "Unauthorized") {
        toast.error("Could not connect to the server.");
      }
    } finally {
      setIsDeleteModalOpen(false);
      setCategoryToDelete(null);
      setConfirmInput("");
    }
  };

  const handleCloseModal = () => {
    setShowModal(false);
    setEditingCategory(null);
    setFormData({
      name: "",
      slug: "",
      description: "",
      parent_id: "",
      meta_title: "",
      meta_description: "",
    });
    setShowSeoFields(false);
  };

  const handleChange = (e) => {
    const { name, value } = e.target;
    setFormData((prev) => ({
      ...prev,
      [name]: value,
      ...(name === "name" && !editingCategory
        ? {
            slug: value
              .toLowerCase()
              .replace(/\s+/g, "-")
              .replace(/[^a-z0-9-]/g, ""),
          }
        : {}),
    }));
  };

  const parentOptions = useMemo(() => {
    return allCategories
      .filter((cat) => {
        if (editingCategory && cat.id === editingCategory.id) return false;
        if (cat.parent_id !== null) return false;
        return true;
      })
      .sort((a, b) => a.name.localeCompare(b.name));
  }, [allCategories, editingCategory]);

  const hasChildren = useMemo(() => {
    if (!editingCategory) return false;
    return allCategories.some((cat) => cat.parent_id === editingCategory.id);
  }, [allCategories, editingCategory]);

  if (!isInitialLoadComplete) {
    return null;
  }

  return (
    <div className="space-y-6 relative">
      {/* --- HEADER CARD --- */}
      <div className="flex items-start gap-4 p-4 rounded-lg bg-indigo-50 dark:bg-indigo-900/20 border border-indigo-100 dark:border-indigo-800/50">
        <div className="p-2 bg-indigo-100 dark:bg-indigo-800 rounded-lg text-indigo-600 dark:text-indigo-300 shrink-0">
          <SafeIcon icon={FiGrid} className="w-5 h-5" />
        </div>
        <div className="flex-1">
          <div className="flex flex-col sm:flex-row sm:items-center justify-between gap-4">
            <div>
              <h4 className="font-bold text-gray-900 dark:text-white">
                Calculator Categories
              </h4>
              <p className="text-sm text-gray-600 dark:text-gray-300 mt-1 leading-relaxed">
                Organize your calculators into a logical hierarchy for better
                user navigation.
              </p>
            </div>
            <motion.button
              whileHover={{ scale: 1.02 }}
              whileTap={{ scale: 0.98 }}
              onClick={() => setShowModal(true)}
              className="flex items-center justify-center space-x-2 bg-indigo-600 text-white px-5 py-2.5 rounded-lg shadow-sm hover:bg-indigo-700 transition-colors font-medium text-sm whitespace-nowrap"
            >
              <SafeIcon icon={FiPlus} className="w-4 h-4" />
              <span>New Category</span>
            </motion.button>
          </div>
        </div>
      </div>

      <div className="bg-white dark:bg-gray-800 rounded-xl shadow-sm border border-gray-200 dark:border-gray-700 overflow-hidden">
        {/* Header Row: Search + Bulk Actions */}
        <div className="px-6 py-3 border-b border-gray-200 dark:border-gray-700 bg-gray-50 dark:bg-gray-700/50 flex flex-col sm:flex-row justify-between items-center gap-4">
          {/* Left: Count */}
          <div className="flex items-center gap-2">
            <div className="bg-white dark:bg-gray-800 px-3 py-1.5 rounded-md border border-gray-200 dark:border-gray-600 text-xs font-bold text-gray-600 dark:text-gray-300 shadow-sm flex items-center gap-2">
              <SafeIcon icon={FiList} className="w-3 h-3 text-indigo-500" />
              {filteredCategories.length} Categories
            </div>
          </div>

          {/* Right: Actions & Search */}
          <div className="flex items-center gap-3 w-full sm:w-auto justify-end">
            {/* Bulk Action */}
            <AnimatePresence>
              {selectedIds.length > 0 && (
                <motion.div
                  initial={{ opacity: 0, x: 10 }}
                  animate={{ opacity: 1, x: 0 }}
                  exit={{ opacity: 0, x: 10 }}
                  className="flex items-center gap-2 pr-3 border-r border-gray-300 dark:border-gray-600"
                >
                  <span className="text-xs font-semibold text-indigo-800 dark:text-indigo-200 whitespace-nowrap hidden md:block">
                    {selectedIds.length} selected
                  </span>
                  <button
                    onClick={handleBulkDelete}
                    disabled={isBulkDeleting}
                    className="text-xs font-bold flex items-center gap-1.5 text-red-600 hover:text-red-800 bg-red-50 dark:bg-red-900/20 border border-red-200 dark:border-red-800/50 px-2 py-1.5 rounded-md transition-colors shadow-sm"
                  >
                    <SafeIcon
                      icon={isBulkDeleting ? FiAlertTriangle : FiTrash2}
                      className="w-3.5 h-3.5"
                    />
                    {isBulkDeleting ? "Deleting..." : "Delete"}
                  </button>
                </motion.div>
              )}
            </AnimatePresence>

            {/* Search Input */}
            <div className="relative w-full sm:w-64">
              <SafeIcon
                icon={FiSearch}
                className="absolute left-3 top-1/2 transform -translate-y-1/2 text-gray-400 w-4 h-4"
              />
              <input
                type="text"
                placeholder="Search categories..."
                value={searchTerm}
                onChange={(e) => setSearchTerm(e.target.value)}
                className="w-full pl-9 pr-4 py-1.5 text-sm border border-gray-300 dark:border-gray-600 rounded-lg focus:ring-2 focus:ring-indigo-500 focus:border-transparent dark:bg-gray-700 dark:text-white transition-all placeholder-gray-400"
              />
            </div>
          </div>
        </div>

        <div className="relative min-h-[200px]">
          <AnimatePresence>
            {isUpdating && (
              <motion.div
                initial={{ opacity: 0 }}
                animate={{ opacity: 1 }}
                exit={{ opacity: 0 }}
                className="absolute inset-0 bg-white/60 dark:bg-gray-900/60 backdrop-blur-[1px] z-10 flex items-center justify-center"
              >
                <LoadingSpinner size="lg" />
              </motion.div>
            )}
          </AnimatePresence>

          <DataTable
            columns={columns}
            data={filteredCategories}
            loading={false}
            onEdit={handleEdit}
            onDelete={handleDelete}
            pagination={false}
            searchable={false}
            selectable={true}
            selectedIds={selectedIds}
            onSelectRow={handleSelectRow}
            onSelectAll={handleSelectAll}
          />
        </div>
      </div>

      {/* --- ADD/EDIT MODAL --- */}
      <Modal
        isOpen={showModal}
        onClose={handleCloseModal}
        title={editingCategory ? "Edit Category" : "New Category"}
        size="lg"
        disableAnimation={true} // --- INSTANT POPUP ---
      >
        <form onSubmit={handleSubmit} className="flex flex-col h-full">
          <div className="flex-1 overflow-y-auto max-h-[60vh] p-1 pr-2 space-y-5 custom-scrollbar">
            {/* 2-Column Grid */}
            <div className="grid grid-cols-1 md:grid-cols-2 gap-5">
              <div>
                <label className="block text-sm font-medium text-gray-700 dark:text-gray-300 mb-1.5">
                  Name <span className="text-red-500">*</span>
                </label>
                <input
                  type="text"
                  name="name"
                  value={formData.name}
                  onChange={handleChange}
                  required
                  className="w-full px-3 py-2 border border-gray-300 dark:border-gray-600 rounded-lg focus:ring-2 focus:ring-calcpro-primary focus:border-transparent dark:bg-gray-700 dark:text-white"
                />
              </div>
              <div>
                <label className="block text-sm font-medium text-gray-700 dark:text-gray-300 mb-1.5">
                  Slug <span className="text-red-500">*</span>
                </label>
                <input
                  type="text"
                  name="slug"
                  value={formData.slug}
                  onChange={handleChange}
                  required
                  className="w-full px-3 py-2 border border-gray-300 dark:border-gray-600 rounded-lg focus:ring-2 focus:ring-calcpro-primary focus:border-transparent dark:bg-gray-700 dark:text-white"
                />
              </div>
            </div>

            <div>
              <label className="block text-sm font-medium text-gray-700 dark:text-gray-300 mb-1.5">
                Parent Category
              </label>
              <div className={`relative ${hasChildren ? "opacity-50" : ""}`}>
                <SafeIcon
                  icon={FiFolder}
                  className="absolute left-3 top-2.5 text-gray-400 w-4 h-4"
                />
                <select
                  name="parent_id"
                  value={hasChildren ? "" : formData.parent_id}
                  onChange={handleChange}
                  disabled={hasChildren}
                  className="w-full pl-10 pr-4 py-2 border border-gray-300 dark:border-gray-600 rounded-lg focus:ring-2 focus:ring-calcpro-primary focus:border-transparent dark:bg-gray-700 dark:text-white disabled:cursor-not-allowed"
                >
                  <option value="">None (Top Level)</option>
                  {!hasChildren &&
                    parentOptions.map((cat) => (
                      <option key={cat.id} value={cat.id}>
                        {cat.name}
                      </option>
                    ))}
                </select>
              </div>
              {hasChildren && (
                <div className="flex items-start gap-2 mt-2 p-2 bg-amber-50 dark:bg-amber-900/20 rounded-md border border-amber-100 dark:border-amber-800/50">
                  <SafeIcon
                    icon={FiAlertCircle}
                    className="w-4 h-4 text-amber-600 shrink-0 mt-0.5"
                  />
                  <span className="text-xs text-amber-700 dark:text-amber-200 leading-tight">
                    Contains sub-categories. Cannot be assigned to another
                    parent (2-level max).
                  </span>
                </div>
              )}
            </div>

            <div>
              <label className="block text-sm font-medium text-gray-700 dark:text-gray-300 mb-1.5">
                Visual Description (Sub-heading)
              </label>
              <textarea
                name="description"
                value={formData.description}
                onChange={handleChange}
                rows={2}
                placeholder="Appears on the public category page."
                className="w-full px-3 py-2 border border-gray-300 dark:border-gray-600 rounded-lg focus:ring-2 focus:ring-calcpro-primary focus:border-transparent dark:bg-gray-700 dark:text-white resize-none"
              />
            </div>

            {/* --- SEO Toggle Section --- */}
            <div className="pt-3 border-t border-gray-100 dark:border-gray-700">
              <button
                type="button"
                onClick={() => setShowSeoFields(!showSeoFields)}
                className="flex items-center gap-2 text-sm font-semibold text-indigo-600 dark:text-indigo-400 hover:text-indigo-700 dark:hover:text-indigo-300 transition-colors w-full justify-between group"
              >
                <span className="flex items-center gap-2">
                  <SafeIcon icon={FiGlobe} className="w-4 h-4" /> SEO Settings
                  (Advanced)
                </span>
                <SafeIcon
                  icon={showSeoFields ? FiChevronUp : FiChevronDown}
                  className="w-4 h-4 text-gray-400 group-hover:text-indigo-500"
                />
              </button>
              <AnimatePresence>
                {showSeoFields && (
                  <motion.div
                    initial={{ opacity: 0, height: 0 }}
                    animate={{ opacity: 1, height: "auto" }}
                    exit={{ opacity: 0, height: 0 }}
                    className="overflow-hidden"
                  >
                    <div className="mt-3 space-y-4 bg-gray-50 dark:bg-gray-700/30 p-4 rounded-lg border border-gray-200 dark:border-gray-600">
                      <div>
                        <label className="block text-xs font-bold text-gray-500 dark:text-gray-400 uppercase mb-1">
                          SEO Title (Meta Title)
                        </label>
                        <input
                          type="text"
                          name="meta_title"
                          value={formData.meta_title}
                          onChange={handleChange}
                          placeholder="Overrides global template"
                          className="w-full px-3 py-2 border border-gray-300 dark:border-gray-600 rounded-lg focus:ring-2 focus:ring-blue-500 dark:bg-gray-700 dark:text-white text-sm"
                        />
                      </div>
                      <div>
                        <label className="block text-xs font-bold text-gray-500 dark:text-gray-400 uppercase mb-1">
                          SEO Description (Meta Description)
                        </label>
                        <textarea
                          name="meta_description"
                          value={formData.meta_description}
                          onChange={handleChange}
                          rows={2}
                          placeholder="Specific text for Google. Defaults to Visual Description if empty."
                          className="w-full px-3 py-2 border border-gray-300 dark:border-gray-600 rounded-lg focus:ring-2 focus:ring-blue-500 dark:bg-gray-700 dark:text-white text-sm resize-none"
                        />
                      </div>
                    </div>
                  </motion.div>
                )}
              </AnimatePresence>
            </div>
          </div>

          {/* Footer Buttons (Sticky) */}
          <div className="pt-4 border-t border-gray-200 dark:border-gray-700 mt-2 flex justify-end gap-3 bg-white dark:bg-gray-800 z-10">
            <button
              type="button"
              onClick={handleCloseModal}
              className="px-4 py-2 text-sm font-medium text-gray-700 dark:text-gray-300 border border-gray-300 dark:border-gray-600 rounded-lg hover:bg-gray-50 dark:hover:bg-gray-700 transition-colors"
            >
              Cancel
            </button>
            <button
              type="submit"
              disabled={isSubmitting}
              className="px-4 py-2 text-sm font-bold bg-calcpro-primary text-white rounded-lg hover:bg-calcpro-primary/90 disabled:opacity-60 flex items-center gap-2 shadow-sm transition-all"
            >
              {isSubmitting ? (
                <>
                  <LoadingSpinner size="xs" color="white" />
                  <span>Saving...</span>
                </>
              ) : (
                <span>
                  {editingCategory ? "Update Category" : "Create Category"}
                </span>
              )}
            </button>
          </div>
        </form>
      </Modal>

      {/* --- DELETE CONFIRMATION MODAL --- */}
      <Modal
        isOpen={isDeleteModalOpen}
        onClose={() => setIsDeleteModalOpen(false)}
        title="Confirm Deletion"
        type="danger"
        size="md"
        disableAnimation={true} // --- INSTANT POPUP ---
      >
        <div className="space-y-4">
          <div className="bg-red-50 dark:bg-red-900/20 p-4 rounded-lg border border-red-200 dark:border-red-800 flex items-start gap-3">
            <SafeIcon
              icon={FiAlertTriangle}
              className="w-6 h-6 text-red-600 shrink-0 mt-0.5"
            />
            <div>
              <h4 className="text-sm font-bold text-red-900 dark:text-red-100">
                Permanent Action
              </h4>
              <p className="text-xs text-red-800 dark:text-red-200/80 mt-1">
                You are about to permanently delete the category:{" "}
                <strong>{categoryToDelete?.name}</strong>. This action cannot be
                undone.
              </p>
            </div>
          </div>
          <div>
            <label className="block text-sm font-medium text-gray-700 dark:text-gray-300 mb-2">
              Type <strong>DELETE</strong> to confirm:
            </label>
            <input
              type="text"
              value={confirmInput}
              onChange={(e) => setConfirmInput(e.target.value)}
              className="w-full px-3 py-2 border border-gray-300 dark:border-gray-600 rounded-md focus:ring-2 focus:ring-red-500 dark:bg-gray-700 dark:text-white"
              placeholder="DELETE"
            />
          </div>
          <div className="flex justify-end gap-3 pt-2">
            <button
              onClick={() => setIsDeleteModalOpen(false)}
              className="px-4 py-2 text-gray-700 dark:text-gray-300 hover:bg-gray-100 dark:hover:bg-gray-700 rounded-lg transition-colors"
            >
              Cancel
            </button>
            <button
              onClick={confirmDelete}
              disabled={confirmInput !== "DELETE"}
              className="px-4 py-2 bg-red-600 text-white rounded-lg hover:bg-red-700 disabled:opacity-50 disabled:cursor-not-allowed transition-colors font-medium shadow-sm"
            >
              Delete Forever
            </button>
          </div>
        </div>
      </Modal>

      {/* --- BULK DELETE CONFIRMATION MODAL --- */}
      <Modal
        isOpen={isBulkDeleteModalOpen}
        onClose={() => setIsBulkDeleteModalOpen(false)}
        title={`Confirm Bulk Deletion`}
        type="danger"
        size="md"
        disableAnimation={true} // --- INSTANT POPUP ---
      >
        <div className="space-y-4">
          <div className="bg-red-50 dark:bg-red-900/20 p-4 rounded-lg border border-red-200 dark:border-red-800 flex items-start gap-3">
            <SafeIcon
              icon={FiAlertTriangle}
              className="w-6 h-6 text-red-600 shrink-0 mt-0.5"
            />
            <div>
              <h4 className="text-sm font-bold text-red-900 dark:text-red-100">
                Permanent Action
              </h4>
              <p className="text-xs text-red-800 dark:text-red-200/80 mt-1">
                You are about to permanently delete{" "}
                <strong>{selectedIds.length} categories</strong>. This cannot be
                undone. Some items may not be deleted if they are assigned to
                content or have sub-categories.
              </p>
            </div>
          </div>
          <div>
            <label className="block text-sm font-medium text-gray-700 dark:text-gray-300 mb-2">
              Type <strong>DELETE</strong> to confirm:
            </label>
            <input
              type="text"
              value={confirmInput}
              onChange={(e) => setConfirmInput(e.target.value)}
              className="w-full px-3 py-2 border border-gray-300 dark:border-gray-600 rounded-md focus:ring-2 focus:ring-red-500 dark:bg-gray-700 dark:text-white"
              placeholder="DELETE"
            />
          </div>
          <div className="flex justify-end gap-3 pt-2">
            <button
              onClick={() => setIsBulkDeleteModalOpen(false)}
              className="px-4 py-2 text-gray-700 dark:text-gray-300 hover:bg-gray-100 dark:hover:bg-gray-700 rounded-lg transition-colors"
            >
              Cancel
            </button>
            <button
              onClick={confirmBulkDelete}
              disabled={isBulkDeleting || confirmInput !== "DELETE"}
              className="px-4 py-2 bg-red-600 text-white rounded-lg hover:bg-red-700 disabled:opacity-50 disabled:cursor-not-allowed transition-colors font-medium shadow-sm flex items-center gap-2"
            >
              {isBulkDeleting ? (
                <>
                  <LoadingSpinner size="xs" color="white" />
                  <span>Deleting...</span>
                </>
              ) : (
                "Delete All"
              )}
            </button>
          </div>
        </div>
      </Modal>
    </div>
  );
};

export default CalculatorCategories;
