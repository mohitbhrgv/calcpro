import React, { useState, useEffect } from "react";
import { Outlet, Navigate, useLocation } from "react-router-dom";
import { useAuth } from "../../context/AuthContext";
import Sidebar from "./Sidebar";
import Header from "./Header";
import LoadingSpinner from "../common/LoadingSpinner";
import { useAdminCache } from "../../context/AdminCacheContext";

const AdminLayout = () => {
  const { user, loading } = useAuth();
  const [sidebarOpen, setSidebarOpen] = useState(true);
  
  // --- Dynamic Identity Logic ---
  const { getCache } = useAdminCache();
  // We still fetch settings for the Site Title, but we won't use the favicon
  const settings = getCache("settings_general");
  const location = useLocation();

  useEffect(() => {
    // 1. Scroll to top on route change
    window.scrollTo(0, 0);

    // 2. Strict Title Format: "Admin Panel | {SiteName}"
    // This is helpful for admins managing multiple sites, so we keep it.
    const siteName = settings?.siteName || "CalcPro"; 
    document.title = `Admin Panel | ${siteName}`;

    // --- FIX: REMOVED DYNAMIC FAVICON INJECTION ---
    // The Admin Panel should strictly use its own static system icon (e.g., cpanel.svg)
    // and should not mirror the public-facing "Site Icon".
    
  }, [location.pathname, settings]); 

  if (loading) {
    return <LoadingSpinner />;
  }

  if (!user) {
    return <Navigate to="/admin/login" replace />;
  }

  return (
    <div className="flex min-h-screen bg-gray-50 dark:bg-gray-900">
      <Sidebar
        isOpen={sidebarOpen}
        onToggle={() => setSidebarOpen(!sidebarOpen)}
      />
      
      <div className="flex-1 flex flex-col min-w-0 transition-all duration-300 ease-in-out">
        <Header onMenuClick={() => setSidebarOpen(!sidebarOpen)} />
        
        {/* Main Content Area */}
        <main className="flex-1 p-6 overflow-x-hidden">
          <div className="max-w-7xl mx-auto">
             <Outlet />
          </div>
        </main>
      </div>
    </div>
  );
};

export default AdminLayout;
